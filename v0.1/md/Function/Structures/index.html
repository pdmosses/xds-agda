    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Structures for types of functions
------------------------------------------------------------------------

-- The contents of this file should usually be accessed from `Function`.

{-# OPTIONS --cubical-compatible --safe #-}

open import Relation.Binary.Core using (Rel)
open import Relation.Binary.Bundles using (Setoid)
open import Relation.Binary.Structures using (IsEquivalence)

module Function.Structures {a b  }
  {A : Set a} (__ : Rel A ) -- Equality over the domain
  {B : Set b} (__ : Rel B ) -- Equality over the codomain
  where

open import Data.Product.Base as Product using (; __; _,_)
open import Function.Base
open import Function.Definitions
open import Level using (__)

------------------------------------------------------------------------
-- One element structures
------------------------------------------------------------------------

record IsCongruent (to : A  B) : Set (a  b    ) where
  field
    cong           : Congruent __ __ to
    isEquivalence : IsEquivalence __
    isEquivalence : IsEquivalence __

  module Eq where

    setoid : Setoid a 
    setoid = record
      { isEquivalence = isEquivalence
      }

    open Setoid setoid public

  module Eq where

    setoid : Setoid b 
    setoid = record
      { isEquivalence = isEquivalence
      }

    open Setoid setoid public


record IsInjection (to : A  B) : Set (a  b    ) where
  field
    isCongruent : IsCongruent to
    injective   : Injective __ __ to

  open IsCongruent isCongruent public


record IsSurjection (f : A  B) : Set (a  b    ) where
  field
    isCongruent : IsCongruent f
    surjective  : Surjective __ __ f

  open IsCongruent isCongruent public

  strictlySurjective : StrictlySurjective __ f
  strictlySurjective x = Product.map ( v  v Eq.refl) (surjective x)


record IsBijection (f : A  B) : Set (a  b    ) where
  field
    isInjection : IsInjection f
    surjective  : Surjective __ __ f

  open IsInjection isInjection public

  bijective : Bijective __ __ f
  bijective = injective , surjective

  isSurjection : IsSurjection f
  isSurjection = record
    { isCongruent = isCongruent
    ; surjective  = surjective
    }

  open IsSurjection isSurjection public
    using (strictlySurjective)


------------------------------------------------------------------------
-- Two element structures
------------------------------------------------------------------------

record IsLeftInverse (to : A  B) (from : B  A) : Set (a  b    ) where
  field
    isCongruent  : IsCongruent to
    from-cong    : Congruent __ __ from
    inverse     : Inverse __ __ to from

  open IsCongruent isCongruent public
    renaming (cong to to-cong)

  strictlyInverse : StrictlyInverse __ to from
  strictlyInverse x = inverse Eq.refl

  isSurjection : IsSurjection to
  isSurjection = record
    { isCongruent = isCongruent
    ; surjective =  y  from y , inverse
    }


record IsRightInverse (to : A  B) (from : B  A) : Set (a  b    ) where
  field
    isCongruent : IsCongruent to
    from-cong   : Congruent __ __ from
    inverse    : Inverse __ __ to from

  open IsCongruent isCongruent public
    renaming (cong to to-cong)

  strictlyInverse : StrictlyInverse __ to from
  strictlyInverse x = inverse Eq.refl


record IsInverse (to : A  B) (from : B  A) : Set (a  b    ) where
  field
    isLeftInverse : IsLeftInverse to from
    inverse      : Inverse __ __ to from

  open IsLeftInverse isLeftInverse public

  isRightInverse : IsRightInverse to from
  isRightInverse = record
    { isCongruent = isCongruent
    ; from-cong   = from-cong
    ; inverse    = inverse
    }

  open IsRightInverse isRightInverse public
    using (strictlyInverse)

  inverse : Inverse __ __ to from
  inverse = inverse , inverse


------------------------------------------------------------------------
-- Three element structures
------------------------------------------------------------------------

record IsBiEquivalence
  (to : A  B) (from : B  A) (from : B  A) : Set (a  b    ) where
  field
    to-isCongruent : IsCongruent to
    from-cong    : Congruent __ __ from
    from-cong    : Congruent __ __ from

  open IsCongruent to-isCongruent public
    renaming (cong to to-cong)


record IsBiInverse
  (to : A  B) (from : B  A) (from : B  A) : Set (a  b    ) where
  field
    to-isCongruent : IsCongruent to
    from-cong     : Congruent __ __ from
    from-cong     : Congruent __ __ from
    inverse       : Inverse __ __ to from
    inverse       : Inverse __ __ to from

  open IsCongruent to-isCongruent public
    renaming (cong to to-cong)

  field
    from : B  A
    isLeftInverse : IsLeftInverse f from

  open IsLeftInverse isLeftInverse public
