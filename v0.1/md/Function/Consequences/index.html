    Function.Consequences

    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Relationships between properties of functions. See
-- `Function.Consequences.Propositional` for specialisations to
-- propositional equality.
------------------------------------------------------------------------

{-# OPTIONS --cubical-compatible --safe #-}

module Function.Consequences where

open import Data.Product.Base as Product
open import Function.Definitions
open import Level using (Level)
open import Relation.Binary.Core using (Rel)
open import Relation.Binary.Bundles using (Setoid)
open import Relation.Binary.Definitions using (Reflexive; Symmetric; Transitive)
open import Relation.Nullary.Negation.Core using (_; contraposition)

private
  variable
    a b   : Level
    A B : Set a
      : Rel A 
    f f : A  B

------------------------------------------------------------------------
-- Injective

contraInjective :  ( : Rel B )  Injective   f 
                   {x y}   ( x y)   ( (f x) (f y))
contraInjective _ inj p = contraposition inj p

------------------------------------------------------------------------
-- Inverse

inversesurjective :  ( : Rel B ) 
                      Inverse   f f 
                      Surjective   f
inversesurjective  inv y = (_ , inv)

------------------------------------------------------------------------
-- Inverse

inverseinjective :  ( : Rel B ) f 
                     Reflexive  
                     Symmetric  
                     Transitive  
                     Inverse   f f 
                     Injective   f
inverseinjective  f refl sym trans inv {x} {y} fxfy =
  trans (sym (inv refl)) (inv fxfy)

------------------------------------------------------------------------
-- Inverse

inversebijective :  ( : Rel B ) 
                     Reflexive  
                     Symmetric  
                     Transitive  
                     Inverse   f f 
                     Bijective   f
inversebijective {f = f}  refl sym trans (inv , inv) =
  (inverseinjective  f refl sym trans inv , inversesurjective  inv)

------------------------------------------------------------------------
-- StrictlySurjective

surjectivestrictlySurjective :  ( : Rel B ) 
                                 Reflexive  
                                 Surjective   f 
                                 StrictlySurjective  f
surjectivestrictlySurjective _ refl surj x =
  Product.map ( v  v refl) (surj x)

strictlySurjectivesurjective : Transitive  
                                 Congruent   f 
                                 StrictlySurjective  f 
                                 Surjective   f
strictlySurjectivesurjective trans cong surj x =
  Product.map ( fyx zy  trans (cong zy) fyx) (surj x)

------------------------------------------------------------------------
-- StrictlyInverse

inversestrictlyInverse :  ( : Rel A ) ( : Rel B ) 
                            Reflexive  
                            Inverse   f f 
                            StrictlyInverse  f f
inversestrictlyInverse _ _ refl sinv x = sinv refl

strictlyInverseinverse : Transitive  
                            Congruent   f 
                            StrictlyInverse  f f 
                            Inverse   f f
strictlyInverseinverse trans cong sinv {x} yfx =
  trans (cong yfx) (sinv x)

------------------------------------------------------------------------
-- StrictlyInverse

inversestrictlyInverse :  ( : Rel A ) ( : Rel B ) 
                            Reflexive  
                            Inverse   f f 
                            StrictlyInverse  f f
inversestrictlyInverse _ _ refl sinv x = sinv refl

strictlyInverseinverse : Transitive  
                            Congruent   f 
                            StrictlyInverse  f f 
                            Inverse   f f
strictlyInverseinverse trans cong sinv {x} yfx =
  trans (cong yfx) (sinv x)
