    Algebra.Construct.NaturalChoice.MinOp

    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Properties of a min operator derived from a spec over a total
-- preorder.
------------------------------------------------------------------------

{-# OPTIONS --cubical-compatible --safe #-}

open import Relation.Binary.Bundles using (TotalPreorder)
open import Algebra.Construct.NaturalChoice.Base using (MinOperator; MinOpMaxOp)

module Algebra.Construct.NaturalChoice.MinOp
  {a  } {O : TotalPreorder a  } (minOp : MinOperator O) where

open import Algebra.Core using (Op)
open import Algebra.Bundles
  using (RawMagma; Magma; Semigroup; Band; CommutativeSemigroup; SelectiveMagma; Monoid)
open import Data.Sum.Base as Sum using (inj; inj; [_,_])
open import Data.Product.Base using (_,_)
open import Function.Base using (id; __)
open import Relation.Binary.Core using (_Preserves__; _Preserves___)
open import Relation.Binary.Definitions using (Maximum; Minimum)

open TotalPreorder O renaming
  ( Carrier   to A
  ; __       to __
  ; -resp-  to -resp-
  ; -resp- to -resp-
  ; -resp- to -resp-
  )
open MinOperator minOp

open import Algebra.Definitions __
open import Algebra.Structures __
open import Relation.Binary.Reasoning.Preorder preorder

------------------------------------------------------------------------
-- Helpful properties

xyx :  x y  x  y  x
xyx x y with total x y
... | inj xy = reflexive (xyxyx xy)
... | inj yx = -resp- (Eq.sym (xyxyy yx)) yx

xyy :  x y  x  y  y
xyy x y with total x y
... | inj xy = -resp- (Eq.sym (xyxyx xy)) xy
... | inj yx = reflexive (xyxyy yx)

------------------------------------------------------------------------
-- Algebraic properties

-comm : Commutative __
-comm x y with total x y
... | inj xy = Eq.trans (xyxyx xy) (Eq.sym (xyxyy xy))
... | inj yx = Eq.trans (xyxyy yx) (Eq.sym (xyxyx yx))

-cong :  x  Congruent (x _)
-cong x {y} {r} yr with total x y
... | inj xy = begin-equality
  x  y    xyxyx xy 
  x       xyxyx (-resp- yr xy) 
  x  r  
... | inj yx = begin-equality
  x  y    xyxyy yx 
  y        yr 
  r       xyxyy (-resp- yr yx) 
  x  r  

-cong :  x  Congruent (_ x)
-cong x {y} {y} yy = begin-equality
  y  x   -comm x y 
  x   y   -cong x yy 
  x   y   -comm x y 
  y  x  

-cong : Congruent __
-cong {x} {x} {y} {y} xx yy = Eq.trans (-cong x yy) (-cong y xx)

-assoc : Associative __
-assoc x y r with total x y | total y r
-assoc x y r | inj xy | inj yr = begin-equality
  (x  y)  r   -cong r (xyxyx xy) 
  x  r         xyxyx (trans xy yr) 
  x             xyxyx xy 
  x  y         -cong x (xyxyx yr) 
  x  (y  r)  
-assoc x y r | inj xy | inj ry = begin-equality
  (x  y)  r   -cong r (xyxyx xy) 
  x  r         -cong x (xyxyy ry) 
  x  (y  r)  
-assoc x y r | inj yx | _ = begin-equality
  (x  y)  r   -cong r (xyxyy yx) 
  y  r         xyxyy (trans (xyx y r) yx) 
  x  (y  r)  

-idem : Idempotent __
-idem x = xyxyx (refl {x})

-sel : Selective __
-sel x y = Sum.map xyxyx xyxyy (total x y)

-identity :  {}  Maximum __   LeftIdentity  __
-identity max = xyxyy  max

-identity :  {}  Maximum __   RightIdentity  __
-identity max = xyxyx  max

-identity :  {}  Maximum __   Identity  __
-identity max = -identity max , -identity max

-zero :  {}  Minimum __   LeftZero  __
-zero min = xyxyx  min

-zero :  {}  Minimum __   RightZero  __
-zero min = xyxyy  min

-zero :  {}  Minimum __   Zero  __
-zero min = -zero min , -zero min

------------------------------------------------------------------------
-- Structures

-isMagma : IsMagma __
-isMagma = record
  { isEquivalence = isEquivalence
  ; -cong        = -cong
  }

-isSemigroup : IsSemigroup __
-isSemigroup = record
  { isMagma = -isMagma
  ; assoc   = -assoc
  }

-isBand : IsBand __
-isBand = record
  { isSemigroup = -isSemigroup
  ; idem        = -idem
  }

-isCommutativeSemigroup : IsCommutativeSemigroup __
-isCommutativeSemigroup = record
  { isSemigroup = -isSemigroup
  ; comm        = -comm
  }

-isSelectiveMagma : IsSelectiveMagma __
-isSelectiveMagma = record
  { isMagma = -isMagma
  ; sel     = -sel
  }

-isMonoid :  {}  Maximum __   IsMonoid __ 
-isMonoid max = record
  { isSemigroup = -isSemigroup
  ; identity    = -identity max
  }

------------------------------------------------------------------------
-- Raw bundles

-rawMagma : RawMagma _ _
-rawMagma = record { __ = __ ; __ = __ }

------------------------------------------------------------------------
-- Bundles

-magma : Magma _ _
-magma = record
  { isMagma = -isMagma
  }

-semigroup : Semigroup _ _
-semigroup = record
  { isSemigroup = -isSemigroup
  }

-band : Band _ _
-band = record
  { isBand = -isBand
  }

-commutativeSemigroup : CommutativeSemigroup _ _
-commutativeSemigroup = record
  { isCommutativeSemigroup = -isCommutativeSemigroup
  }

-selectiveMagma : SelectiveMagma _ _
-selectiveMagma = record
  { isSelectiveMagma = -isSelectiveMagma
  }

-monoid :  {}  Maximum __   Monoid a 
-monoid max = record
  { isMonoid = -isMonoid max
  }

------------------------------------------------------------------------
-- Other properties

xyxxy :  {x y}  x  y  x  x  y
xyxxy {x} {y} xyx with total x y
... | inj xy = xy
... | inj yx = reflexive (Eq.trans (Eq.sym xyx) (xyxyy yx))

xyyyx :  {x y}  x  y  y  y  x
xyyyx {x} {y} xyy = xyxxy (begin-equality
  y  x   -comm y x 
  x  y   xyy 
  y      )

mono--distrib- :  {f}  f Preserves __  __  f Preserves __  __ 
                    x y  f (x  y)  f x  f y
mono--distrib- {f} cong mono x y with total x y
... | inj xy = begin-equality
  f (x  y)   cong (xyxyx xy) 
  f x         xyxyx (mono xy) 
  f x  f y  
... | inj yx = begin-equality
  f (x  y)   cong (xyxyy yx) 
  f y         xyxyy (mono yx) 
  f x  f y  

xyxzy :  {x y} z  x  y  x  z  y
xyxzy z xy = trans (xyx _ z) xy

xyzxy :  {x y} z  x  y  z  x  y
xyzxy y xy = trans (xyy y _) xy

xyzxy :  {x} y z  x  y  z  x  y
xyzxy y z xyz = trans xyz (xyx y z)

xyzxz :  {x} y z  x  y  z  x  z
xyzxz y z xyz = trans xyz (xyy y z)

-mono- : __ Preserves __  __  __
-mono- {x} {y} {u} {v} xy uv with -sel y v
... | inj yvy = -resp- (Eq.sym yvy) (trans (xyx x u) xy)
... | inj yvv = -resp- (Eq.sym yvv) (trans (xyy x u) uv)

-mono- :  x  (_ x) Preserves __  __
-mono- x yz = -mono- yz (refl {x})

-mono- :  x  (x _) Preserves __  __
-mono- x yz = -mono- (refl {x}) yz

-glb :  {x y z}  x  y  x  z  x  y  z
-glb {x} xy xz = -resp- (-idem x) (-mono- xy xz)

-triangulate :  x y z  x  y  z  (x  y)  (y  z)
-triangulate x y z = begin-equality
  x  y  z            -cong z (-cong x (-idem y)) 
  x  (y  y)  z       -assoc x _ _ 
  x  ((y  y)  z)     -cong x (-assoc y y z) 
  x  (y  (y  z))    -assoc x y (y  z) 
  (x  y)  (y  z)
