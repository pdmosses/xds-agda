    Data.Fin.Base

    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Finite sets
------------------------------------------------------------------------

-- Note that elements of Fin n can be seen as natural numbers in the
-- set {m | m < n}. The notation "m" in comments below refers to this
-- natural number view.
{-# OPTIONS --cubical-compatible --safe #-}

module Data.Fin.Base where

open import Data.Nat.Base as  using (; zero; suc)
open import Data.Product.Base as Product using (__; _,_; proj; proj)
open import Data.Sum.Base as Sum using (__; inj; inj; [_,_])
open import Function.Base using (id; __; _on_; flip; _$_)
open import Level using (0)
open import Relation.Binary.Core using (Rel)
open import Relation.Binary.PropositionalEquality.Core using (__; __; refl; cong)
open import Relation.Binary.Indexed.Heterogeneous.Core using (IRel)
open import Relation.Nullary.Negation.Core using (_; contradiction)

private
  variable
    m n : 

------------------------------------------------------------------------
-- Types

-- Fin n is a type with n elements.

data Fin :   Set where
  zero : Fin (suc n)
  suc  : (i : Fin n)  Fin (suc n)

-- A conversion: to "i" = i.

to : Fin n  
to zero    = 0
to (suc i) = suc (to i)

-- A Fin-indexed variant of Fin.

Fin : Fin n  Set
Fin i = Fin (to i)

------------------------------------------------------------------------
-- A cast that actually computes on constructors (as opposed to subst)

cast : .(m  n)  Fin m  Fin n
cast {zero}  {zero}  eq k       = k
cast {suc m} {suc n} eq zero    = zero
cast {suc m} {suc n} eq (suc k) = suc (cast (cong .pred eq) k)

------------------------------------------------------------------------
-- Conversions

-- to is defined above.

-- from n = "n".

from : (n : )  Fin (suc n)
from zero    = zero
from (suc n) = suc (from n)

-- from< {m} _ = "m".

from< : .(m .< n)  Fin n
from< {zero}  {suc _} _   = zero
from< {suc m} {suc _} m<n = suc (from< (.s<s m<n))

-- from< m _ = "m".

from< :  m {n}  .(m .< n)  Fin n
from< zero    {suc _} _    = zero
from< (suc m) {suc _} m<n = suc (from< m (.s<s m<n))

-- canonical liftings of i:Fin m to larger index

-- injection on the left: "i"  n = "i" in Fin (m + n)
infixl 5 __
__ :  {m}  Fin m   n  Fin (m .+ n)
zero     n = zero
(suc i)  n = suc (i  n)

-- injection on the right: n  "i" = "n + i" in Fin (n + m)
infixr 5 __
__ :  {m} n  Fin m  Fin (n .+ m)
zero     i = i
(suc n)  i = suc (n  i)

-- reduce "m + i" _ = "i".

reduce :  (i : Fin (m .+ n))  .(m . to i)  Fin n
reduce {zero}  i       _   = i
reduce {suc _} (suc i) mi = reduce i (.ss mi)

-- inject m "i" = "i".

inject :  {i : Fin n}  Fin i  Fin n
inject {i = suc i} zero    = zero
inject {i = suc i} (suc j) = suc (inject j)

inject! :  {i : Fin (suc n)}  Fin i  Fin n
inject! {n = suc _} {i = suc _} zero    = zero
inject! {n = suc _} {i = suc _} (suc j) = suc (inject! j)

inject : Fin n  Fin (suc n)
inject zero    = zero
inject (suc i) = suc (inject i)

inject : Fin m  .(m . n)  Fin n
inject {n = suc _} zero    _   = zero
inject {n = suc _} (suc i) mn = suc (inject i (.ss mn))

-- lower "i" _ = "i".

lower :  (i : Fin (suc n))  n  to i  Fin n
lower {zero}  zero    ne = contradiction refl ne
lower {suc n} zero    _  = zero
lower {suc n} (suc i) ne = suc (lower i (ne  cong suc))

lower :  (i : Fin m)  .(to i .< n)  Fin n
lower {n = suc n} zero    leq = zero
lower {n = suc n} (suc i) leq = suc (lower i (.ss leq))

-- A strengthening injection into the minimal Fin fibre.
strengthen :  (i : Fin n)  Fin (suc i)
strengthen zero    = zero
strengthen (suc i) = suc (strengthen i)

-- splitAt m "i" = inj "i"      if i < m
--                 inj "i - m"  if i  m
-- This is dual to splitAt from Data.Vec.

splitAt :  m {n}  Fin (m .+ n)  Fin m  Fin n
splitAt zero    i       = inj i
splitAt (suc m) zero    = inj zero
splitAt (suc m) (suc i) = Sum.map suc (splitAt m i)

-- inverse of above function
join :  m n  Fin m  Fin n  Fin (m .+ n)
join m n = [ _ n , m _ ]

-- quotRem k "i" = "i % k" , "i / k"
-- This is dual to group from Data.Vec.

quotRem :  n  Fin (m .* n)  Fin n  Fin m
quotRem {suc m} n i =
  [ (_, zero)
  , Product.map suc  quotRem {m} n
  ] $ splitAt n i

-- a variant of quotRem the type of whose result matches the order of multiplication
remQuot :  n  Fin (m .* n)  Fin m  Fin n
remQuot i = Product.swap  quotRem i

quotient :  n  Fin (m .* n)  Fin m
quotient n = proj  remQuot n

remainder :  n  Fin (m .* n)  Fin n
remainder {m} n = proj  remQuot {m} n

-- inverse of remQuot
combine : Fin m  Fin n  Fin (m .* n)
combine {suc m} {n} zero    j = j  (m .* n)
combine {suc m} {n} (suc i) j = n  (combine i j)

-- Next in progression after splitAt and remQuot
finToFun : Fin (m .^ n)  (Fin n  Fin m)
finToFun {m} {suc n} i zero    = quotient (m .^ n) i
finToFun {m} {suc n} i (suc j) = finToFun (remainder {m} (m .^ n) i) j

-- inverse of above function
funToFin : (Fin m  Fin n)  Fin (n .^ m)
funToFin {zero}  f = zero
funToFin {suc m} f = combine (f zero) (funToFin (f  suc))

------------------------------------------------------------------------
-- Operations

-- Folds.

fold :  {t} (T :   Set t) {m} 
       ( {n}  T n  T (suc n)) 
       ( {n}  T (suc n)) 
       Fin m  T m
fold T f x zero    = x
fold T f x (suc i) = f (fold T f x i)

fold :  {n t} (T : Fin (suc n)  Set t) 
        ( i  T (inject i)  T (suc i)) 
        T zero 
         i  T i
fold             T f x zero     = x
fold {n = suc n} T f x (suc i)  =
  f i (fold (T  inject) (f  inject) x i)

-- Lifts functions.

lift :  k  (Fin m  Fin n)  Fin (k .+ m)  Fin (k .+ n)
lift zero    f i       = f i
lift (suc k) f zero    = zero
lift (suc k) f (suc i) = suc (lift k f i)

-- "i" + "j" = "i + j".

infixl 6 _+_

_+_ :  (i : Fin m) (j : Fin n)  Fin (to i .+ n)
zero  + j = j
suc i + j = suc (i + j)

-- "i" - "j" = "i  j".

infixl 6 _-_

_-_ :  (i : Fin n) (j : Fin (suc i))  Fin (n . to j)
i     - zero   = i
suc i - suc j  = i - j

-- m - "i" = "m  i".

infixl 6 _-_

_-_ : (n : ) (j : Fin (suc n))  Fin (suc n . to j)
n     - zero   = from n
suc n - suc i  = n - i

-- m - "i" = m  i.

infixl 6 _-_

_-_ : (n : )  Fin (suc n)  
n     - zero   = n
suc n - suc i  = n - i

-- pred "i" = "pred i".

pred : Fin n  Fin n
pred zero    = zero
pred (suc i) = inject i

-- opposite "i" = "n - i" (i.e. the additive inverse).

opposite : Fin n  Fin n
opposite {suc n} zero    = from n
opposite {suc n} (suc i) = inject (opposite i)

-- The function f(i,j) = if j>i then j-1 else j
-- This is a variant of the thick function from Conor
-- McBride's "First-order unification by structural recursion".

punchOut :  {i j : Fin (suc n)}  i  j  Fin n
punchOut {_}     {zero}   {zero}  ij = contradiction refl ij
punchOut {_}     {zero}   {suc j} _   = j
punchOut {suc _} {suc i}  {zero}  _   = zero
punchOut {suc _} {suc i}  {suc j} ij = suc (punchOut (ij  cong suc))

-- The function f(i,j) = if ji then j+1 else j

punchIn : Fin (suc n)  Fin n  Fin (suc n)
punchIn zero    j       = suc j
punchIn (suc i) zero    = zero
punchIn (suc i) (suc j) = suc (punchIn i j)

-- The function f(i,j) such that f(i,j) = if ji then j else j-1

pinch : Fin n  Fin (suc n)  Fin n
pinch {suc n} _       zero    = zero
pinch {suc n} zero    (suc j) = j
pinch {suc n} (suc i) (suc j) = suc (pinch i j)

------------------------------------------------------------------------
-- Order relations

infix 4 __ __ _<_ _>_ __ __

__ : IRel Fin 0
i  j = to i . to j

__ : IRel Fin 0
i  j = to i . to j

_<_ : IRel Fin 0
i < j = to i .< to j

_>_ : IRel Fin 0
i > j = to i .> to j

__ :  {n}  Rel (Fin n) 0
i  j =  (i  j)

__ :  {n}  Rel (Fin n) 0
i  j =  (i < j)

------------------------------------------------------------------------
-- An ordering view.

data Ordering {n : } : Fin n  Fin n  Set where
  less    :  greatest (least : Fin greatest) 
            Ordering (inject least) greatest
  equal   :  i  Ordering i i
  greater :  greatest (least : Fin greatest) 
            Ordering greatest (inject least)

compare :  (i j : Fin n)  Ordering i j
compare zero    zero    = equal   zero
compare zero    (suc j) = less    (suc j) zero
compare (suc i) zero    = greater (suc i) zero
compare (suc i) (suc j) with compare i j
... | less    greatest least = less    (suc greatest) (suc least)
... | greater greatest least = greater (suc greatest) (suc least)
... | equal   i              = equal   (suc i)
