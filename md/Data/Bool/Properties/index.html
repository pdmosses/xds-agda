-- Properties of __

infix 4 __

__ : DecidableEquality Bool
true   true  = yes refl
false  false = yes refl
true   false = no ()
false  true  = no ()

-setoid : Setoid 0 0
-setoid = setoid Bool

-decSetoid : DecSetoid 0 0
-decSetoid = decSetoid __

------------------------------------------------------------------------
-- Properties of __

-- Relational properties

-reflexive : __  __
-reflexive refl = bb

-refl : Reflexive __
-refl = -reflexive refl

-trans : Transitive __
-trans bb p   = p
-trans ft bb = ft

-antisym : Antisymmetric __ __
-antisym bb _ = refl

-minimum : Minimum __ false
-minimum false = bb
-minimum true  = ft

-maximum : Maximum __ true
-maximum false = ft
-maximum true  = bb

-total : Total __
-total false b = inj (-minimum b)
-total true  b = inj (-maximum b)

infix 4 _?_

_?_ : Decidable __
false ? b     = yes (-minimum b)
true  ? false = no  ()
true  ? true  = yes bb

-irrelevant : Irrelevant __
-irrelevant {_}     ft ft = refl
-irrelevant {false} bb bb = refl
-irrelevant {true}  bb bb = refl

------------------------------------------------------------------------
-- Properties of _<_

-- Relational properties

<-irrefl : Irreflexive __ _<_
<-irrefl refl ()

<-asym : Asymmetric _<_
<-asym f<t ()

<-trans : Transitive _<_
<-trans f<t ()

<-trans : Trans __ _<_ _<_
<-trans bb f<t = f<t

<-trans : Trans _<_ __ _<_
<-trans f<t bb = f<t

<-cmp : Trichotomous __ _<_
<-cmp false false = tri (()) refl  (())
<-cmp false true  = tri< f<t   (()) (())
<-cmp true  false = tri> (()) (()) f<t
<-cmp true  true  = tri (()) refl  (())

infix 4 _<?_

_<?_ : Decidable _<_
false <? false = no  (())
false <? true  = yes f<t
true  <? _     = no  (())

<-resp- : _<_ Respects __
<-resp- = subst (_ <_) , subst (_< _)

<-irrelevant : Irrelevant _<_
<-irrelevant f<t f<t = refl

<-wellFounded : WellFounded _<_
<-wellFounded _ = acc <-acc
  where
    <-acc :  {x y}  y < x  Acc _<_ y
    <-acc f<t = acc  ()

  }

<-isStrictTotalOrder : IsStrictTotalOrder __ _<_
<-isStrictTotalOrder = record

------------------------------------------------------------------------
-- Properties of not

not-involutive : Involutive not
not-involutive true  = refl
not-involutive false = refl

not-injective :  {x y}  not x  not y  x  y
not-injective {false} {false} nxny = refl
not-injective {true}  {true}  nxny = refl

not- :  {x y}  x  y  x  not y
not- {true}  refl ()
not- {false} refl ()

-not :  {x y}  x  y  x  not y
-not {true}  {true}  xy = -elim (xy refl)
-not {true}  {false} _   = refl
-not {false} {true}  _   = refl
-not {false} {false} xy = -elim (xy refl)

-- Properties of _xor_

xor-is-ok :  x y  x xor y  (x  y)  not (x  y)
xor-is-ok true  y = refl
xor-is-ok false y = sym (-identity _)

true-xor :  x  true xor x  not x
true-xor false = refl
true-xor true  = refl

xor-same :  x  x xor x  false
xor-same false = refl
xor-same true  = refl

not-distrib-xor :  x y  not (x xor y)  (not x) xor y
not-distrib-xor false y = refl
not-distrib-xor true  y = not-involutive _

not-distrib-xor :  x y  not (x xor y)  x xor (not y)
not-distrib-xor false y = refl
not-distrib-xor true  y = refl

xor-assoc : Associative _xor_
xor-assoc true  y z = sym (not-distrib-xor y z)
xor-assoc false y z = refl

xor-comm : Commutative _xor_
xor-comm false false = refl
xor-comm false true  = refl
xor-comm true  false = refl
xor-comm true  true  = refl

xor-identity : LeftIdentity false _xor_
xor-identity _ = refl

xor-identity : RightIdentity false _xor_
xor-identity false = refl
xor-identity true  = refl

xor-identity : Identity false _xor_
xor-identity = xor-identity , xor-identity

xor-inverse : LeftInverse true not _xor_
xor-inverse false = refl
xor-inverse true = refl

xor-inverse : RightInverse true not _xor_
xor-inverse x = xor-comm x (not x)  trans  xor-inverse x

xor-inverse : Inverse true not _xor_
xor-inverse = xor-inverse , xor-inverse

-distrib-xor : __ DistributesOver _xor_
-distrib-xor false y z = refl
-distrib-xor true  y z = refl

-distrib-xor x false z    = refl

-distrib-xor = -distrib-xor , -distrib-xor

-- Properties of T

open Relation.Nullary.Decidable.Core public using (T?)

T- :  {x}  T x  x  true
T- {false} = mk ( ())       ( ())
T- {true}  = mk (const refl) (const _)

T-not- :  {x}  T (not x)  x  false
T-not- {false} = mk (const refl) (const _)
T-not- {true}  = mk ( ())       ( ())

T- :  {x y}  T (x  y)  (T x  T y)
T- {true}  {true}  = mk (const (_ , _)) (const _)
T- {true}  {false} = mk ( ())          proj
T- {false} {_}     = mk ( ())          proj

T- :  {x y}  T (x  y)  (T x  T y)
T- {true}  {_}     = mk inj (const _)
T- {false} {true}  = mk inj (const _)
T- {false} {false} = mk inj [ id , id ]

T-irrelevant : U.Irrelevant T
T-irrelevant {true}  _  _  = refl

T?-diag :  b  T b  True (T? b)
T?-diag b = fromWitness

-- Miscellaneous other properties

 : {x y z : Bool}  x  z  y  z  x  y
 {true } {true }         hyp = refl
 {true } {false} {true } hyp = sym (Equivalence.to hyp refl)
 {true } {false} {false} hyp = Equivalence.from hyp refl
 {false} {true } {true } hyp = Equivalence.from hyp refl
 {false} {true } {false} hyp = sym (Equivalence.to hyp refl)
 {false} {false}         hyp = refl
