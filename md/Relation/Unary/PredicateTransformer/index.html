    Relation.Unary.PredicateTransformer

    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Predicate transformers
------------------------------------------------------------------------

{-# OPTIONS --cubical-compatible --safe #-}

module Relation.Unary.PredicateTransformer where

open import Data.Product.Base using ()
open import Function.Base using (__)
open import Level hiding (__)
open import Relation.Unary
open import Relation.Binary.Core using (REL)

private
  variable
    a b c i     : Level
    A : Set a
    B : Set b
    C : Set c

------------------------------------------------------------------------
-- Heterogeneous and homogeneous predicate transformers

PT : Set a  Set b  (  : Level)  Set _
PT A B   = Pred A   Pred B 

Pt : Set a  ( : Level)  Set _
Pt A  = PT A A  

------------------------------------------------------------------------
-- Composition and identity

infixr 9 __

__ : PT B C    PT A B    PT A C  _
S  T = S  T

skip : PT A A  
skip P = P

------------------------------------------------------------------------
-- Operations on predicates extend pointwise to predicate transformers

-- The bottom and the top of the predicate transformer lattice.

abort : PT A B 0 0
abort =  _  

magic : PT A B 0 0
magic =  _  U

-- Negation.

infix 8 _

_ : PT A B    PT A B  
 T =   T

-- Refinement.

infix 4 __ __ __ __

__ : PT A B    PT A B    Set _
S  T =  {X}  S X  T X

__ : PT A B    PT A B    Set _
S  T =  X  S X  T X

__ : PT A B    PT A B    Set _
T  S = T  S

__ : PT A B    PT A B    Set _
T  S = S  T

-- The dual of refinement.

infix 4 __

__ : PT A B    PT A B    Set _
S  T =   X  S X  T X

-- Union.

infixl 6 __

__ : PT A B    PT A B    PT A B  
S  T =  X  S X  T X

-- Intersection.

infixl 7 __

__ : PT A B    PT A B    PT A B  
S  T =  X  S X  T X

-- Implication.

infixl 8 __

__ : PT A B    PT A B    PT A B  
S  T =  X  S X  T X

-- Infinitary union and intersection.

infix 9  

 :  (I : Set i)  (I  PT A B  )  PT A B  _
 I T =  X  [ i  I ] T i X

syntax  I ( i  T) = [ i  I ] T

 :  (I : Set i)  (I  PT A B  )  PT A B  _
 I T =  X  [ i  I ] T i X

syntax  I ( i  T) = [ i  I ] T

-- Angelic and demonic update.

_ : REL A B   PT B A  _
 R  P =  x  R x  P

[_] : REL A B   PT B A  _
[ R ] P =  x  R x  P
