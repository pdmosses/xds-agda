-- The Agda standard library
--
-- Properties of constructions over unary relations
------------------------------------------------------------------------

{-# OPTIONS --cubical-compatible --safe #-}

module Relation.Unary.Properties where

open import Data.Product.Base as Product using (__; _,_; swap; proj; zip)
open import Data.Sum.Base using (inj; inj)
open import Data.Unit.Base using (tt)
open import Level using (Level)
open import Relation.Binary.Core as Binary
open import Relation.Binary.Definitions
  hiding (Decidable; Universal; Irrelevant; Empty)
open import Relation.Binary.PropositionalEquality.Core using (refl; __)
open import Relation.Unary
open import Relation.Nullary.Decidable as Dec using (yes; no; _-dec_; _-dec_; ?; map; does)
open import Function.Base using (id; _$_; __)

private
  variable
    a b     : Level
    A : Set a
    B : Set b

------------------------------------------------------------------------
-- The empty set

? : Decidable {A = A} 
? _ = no ()

-Empty : Empty {A = A} 
-Empty x ()

-Universal : Universal {A = A} ( )
-Universal =  x x  x

------------------------------------------------------------------------
-- The universe

U? : Decidable {A = A} U
U? _ = yes tt

U-Universal : Universal {A = A} U
U-Universal =  _  _

U-Empty : Empty {A = A} ( U)
U-Empty =  x xU  xU _

------------------------------------------------------------------------
-- Subset properties

- : (P : Pred A )    P
- P ()

-U : (P : Pred A )  P  U
-U P _ = _

-refl : Reflexive {A = Pred A } __
-refl xP = xP

-reflexive : Binary.__ {A = Pred A } {B = Pred A } __ __
-reflexive (PQ , QP) = PQ

-trans : Trans {A = Pred A } {B = Pred A } {C = Pred A } __ __ __
-trans PQ QR xP = QR (PQ xP)

-antisym : Antisymmetric {A = Pred A } __ __
-antisym = _,_

-min : Min {B = Pred A } __ 
-min = -

-max : Max {A = Pred A } __ U
-max = -U

 : Binary.__ {A = Pred A } {B = Pred A } __ __
 = proj

-trans : Trans {A = Pred A } {B = Pred A } {C = Pred A } __ __ __
-trans (PQ , QP) (QR , RQ) = ( xP  QR (PQ xP)) , ( RP  RQ ( xR  PQ (RP xR)))

--trans : Trans {A = Pred A } {B = Pred A } {C = Pred A } __ __ __
--trans (PQ , QP) QR = ( xP  QR (PQ xP)) , ( RP  QP ( xQ  RP (QR xQ)))

--trans : Trans {A = Pred A } {B = Pred A } {C = Pred A } __ __ __
--trans PQ (QR , RQ) = ( xP  QR (PQ xP)) , ( RP  RQ ( RQ  PQ (RP RQ)))

-resp- : _Respects_ {A = Pred A } {B = Pred A } __ __
-resp- (QR , _) PQ = --trans PQ QR

-resp- : _Respects_ {A = Pred A } {B = Pred A } __ __
-resp- (_ , RQ) PQ = --trans RQ PQ

-resp- : _Respects_ {A = Pred A } __ __
-resp- = -resp- , -resp-

-irrefl : Irreflexive {A = Pred A } {B = Pred A } __ __
-irrefl (_ , QP) (_ , QP) = QP QP

-antisym : Antisymmetric {A = Pred A } __ __
-antisym (PQ , _) (QP , _) = -antisym PQ QP

-asym : Asymmetric {A = Pred A } __
-asym (_ , QP) = QP  proj

- : (P : Pred A )    P
- _ _ =  ()

-U : (P : Pred A )  P  U
-U _ _ _ = _

-refl : Reflexive {A = Pred A } __
-refl x xP = xP

-reflexive (PQ , QP) = PQ

-max = -U

-- Equality properties

-refl : Reflexive {A = Pred A } __
-refl = id , id

-sym : Sym {A = Pred A } {B = Pred A } __ __
-sym = swap

-trans : Trans {A = Pred A } {B = Pred A } {C = Pred A } __ __ __
-trans = zip ( PQ QR xP  QR (PQ xP)) ( QP RQ xR  QP (RQ xR))

-refl : Reflexive {A = Pred A } __
-refl = ( _  id) , ( _  id)

-sym : Sym {A = Pred A } {B = Pred A } __ __
-sym = swap

-trans : Trans {A = Pred A } {B = Pred A } {C = Pred A } __ __ __
-trans = zip ( PQ QR x xP  QR x (PQ x xP))  QP RQ x xR  QP x (RQ x xR)

 : Binary.__ {A = Pred A } {B = Pred A } __ __
 = Product.map  

 : Binary.__ {A = Pred A } {B = Pred A } __ __
 = Product.map  

------------------------------------------------------------------------
-- Decidability properties

map : {P : Pred A } {Q : Pred A } 
      P  Q  Decidable P  Decidable Q
map (PQ , QP) P? x = map PQ QP (P? x)

? : {P : Pred A }  Decidable P  Decidable ( P)
? P? x = ? (P? x)

infix 2 _?_ _?_
infix 10 _~?
infixr 1 _?_
infixr 7 _?_
infixr 6 _?_

_?_ : {P : Pred A } {Q : Pred A } 
       Decidable P  Decidable Q  Decidable (P  Q)
_?_ P? Q? x = (P? x) -dec (Q? x)

_?_ : {P : Pred A } {Q : Pred A } 
       Decidable P  Decidable Q  Decidable (P  Q)
_?_ P? Q? x = (P? x) -dec (Q? x)

_?_ : {P : Pred A } {Q : Pred B } 
       Decidable P  Decidable Q  Decidable (P  Q)
_?_ P? Q? (a , b) = (P? a) -dec (Q? b)

_?_ : {P : Pred A } {Q : Pred B } 
       Decidable P  Decidable Q  Decidable (P  Q)
_?_ P? Q? (a , b) = (P? a) -dec (Q? b)

_?_ : {P : Pred A } {Q : Pred B } 
       Decidable P  Decidable Q  Decidable (P  Q)
_?_ P? Q? (inj a) = P? a
_?_ P? Q? (inj b) = Q? b

_~? : {P : Pred (A  B) }  Decidable P  Decidable (P ~)
_~? P? = P?  swap

does- : {P : Pred A }  (P? P? : Decidable P) 
         does  P?  does  P?
does- P? P? x = Dec.does- (P? x) (P? x)

does- : {P : Pred A } {Q : Pred A }  P  Q 
         (P? : Decidable P)  (Q? : Decidable Q) 
         does  P?  does  Q?
does- PQ P? = does- (map PQ P?)

------------------------------------------------------------------------
-- Irrelevant properties

U-irrelevant : Irrelevant {A = A} U
U-irrelevant a b = refl

-irrelevant : (P : Pred A )  Irrelevant ( P)
-irrelevant P a b = refl
