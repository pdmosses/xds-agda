    Relation.Binary.Construct.Composition

    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Composition of two binary relations
------------------------------------------------------------------------

{-# OPTIONS --cubical-compatible --safe #-}

module Relation.Binary.Construct.Composition where

open import Data.Product.Base using (; __; _,_)
open import Function.Base using (__)
open import Level using (Level; __)
open import Relation.Binary.Core using (Rel; REL; __)
open import Relation.Binary.Structures using (IsPreorder)
open import Relation.Binary.Definitions
  using (_Respects_; _Respects_; _Respects_; _Respects_
        ; Reflexive; Transitive)

private
  variable
    a b c    : Level
    A : Set a
    B : Set b
    C : Set c

------------------------------------------------------------------------
-- Definition

infixr 9 __

__ : {A : Set a} {B : Set b} {C : Set c} 
      REL A B   REL B C   REL A C (b    )
L  R =  i j    k  L i k  R k j

------------------------------------------------------------------------
-- Properties

module _ (L : Rel A ) (R : Rel A ) where

  reflexive : Reflexive L  Reflexive R  Reflexive (L  R)
  reflexive L-refl R-refl = _ , L-refl , R-refl

  respects :  {p} {P : A  Set p} 
             P Respects L  P Respects R  P Respects (L  R)
  respects resp-L resp-R (k , Lik , Rkj) = resp-R Rkj  resp-L Lik

module _ { : Rel A } (L : REL A B ) (R : REL B C ) where

  respects : L Respects   (L  R) Respects 
  respects L ii (k , Lik , Rkj) = k , L ii Lik , Rkj

module _ { : Rel C } (L : REL A B ) (R : REL B C ) where

  respects : R Respects   (L  R) Respects 
  respects R jj (k , Lik , Rkj) = k , Lik , R jj Rkj

module _ { : Rel A } (L : REL A B ) (R : REL B A ) where

  respects :  L Respects   R Respects   (L  R) Respects 
  respects L R = respects L R R , respects L R L

module _ { : REL A B } (L : REL A B ) (R : Rel B ) where

  implies : Reflexive R  (  L)  (  L  R)
  implies R-refl L {i} {j} ij = j , L ij , R-refl

module _ { : REL A B } (L : Rel A ) (R : REL A B ) where

  implies : Reflexive L  (  R)  (  L  R)
  implies L-refl R {i} {j} ij = i , L-refl , R ij

module _ (L : Rel A ) (R : Rel A ) (comm : R  L  L  R) where

  transitive : Transitive L  Transitive R  Transitive (L  R)
  transitive L-trans R-trans {i} {j} {k} (x , Lix , Rxj) (y , Ljy , Ryk) =
    let z , Lxz , Rzy = comm (j , Rxj , Ljy) in z , L-trans Lix  Lxz , R-trans Rzy Ryk

  isPreorder : { : Rel A }  IsPreorder  L  IsPreorder  R  IsPreorder  (L  R)
  isPreorder O O = record
    { isEquivalence = O.isEquivalence
    ; reflexive     = implies L R O.refl O.reflexive
    ; trans         = transitive O.trans O.trans
    }
    where module O = IsPreorder O; module O = IsPreorder O

transitive : ( : Rel A )  Transitive   (  )  
transitive _ trans (_ , l , r) = trans l r
