    Relation.Binary.Core

    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Properties of binary relations
------------------------------------------------------------------------

-- The contents of this module should be accessed via `Relation.Binary`.

{-# OPTIONS --cubical-compatible --safe #-}

module Relation.Binary.Core where

open import Data.Product.Base using (__)
open import Function.Base using (_on_)
open import Level using (Level; __; suc)

private
  variable
    a b c     : Level
    A : Set a
    B : Set b
    C : Set c

------------------------------------------------------------------------
-- Definitions
------------------------------------------------------------------------

-- Heterogeneous binary relations

REL : Set a  Set b  ( : Level)  Set (a  b  suc )
REL A B  = A  B  Set 

-- Homogeneous binary relations

Rel : Set a  ( : Level)  Set (a  suc )
Rel A  = REL A A 

------------------------------------------------------------------------
-- Relationships between relations
------------------------------------------------------------------------

infix 4 __ __ _=[_]_

-- Implication/containment - could also be written __.
-- and corresponding notion of equivalence

__ : REL A B   REL A B   Set _
P  Q =  {x y}  P x y  Q x y

__ : REL A B   REL A B   Set _
P  Q = P  Q  Q  P

-- Generalised implication - if P  Q it can be read as "f preserves P".

_=[_]_ : Rel A   (A  B)  Rel B   Set _
P =[ f ] Q = P  (Q on f)

-- A synonym for _=[_]_.

_Preserves__ : (A  B)  Rel A   Rel B   Set _
f Preserves P  Q = P =[ f ] Q

-- A binary variant of _Preserves__.

_Preserves___ : (A  B  C)  Rel A   Rel B   Rel C   Set _
__ Preserves P  Q  R =  {x y u v}  P x y  Q u v  R (x  u) (y  v)
