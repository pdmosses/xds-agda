-- Pointwise equality

_-setoid_ :  (A : Set a) (B : Set b)  Setoid _ _
A -setoid B = -setoid A (Trivial.indexedSetoid (setoid B))

:-to- :  {A : Set a} {B : IndexedSetoid A b } 
          ((x : A)  IndexedSetoid.Carrier B x) 
          Dependent.Func (setoid A) B
:-to- {B = B} f = record
  { to = f
  ; cong =  { refl  IndexedSetoid.refl B }
  }

-to- :  {A : Set a} {B : Setoid b } 
         (A  Setoid.Carrier B) 
         Dependent.Func (setoid A) (Trivial.indexedSetoid B)
-to- = :-to-

-- More complex rearrangement lemmas

-- A lemma that is very similar to Lemma2.4.3 from the HoTT book.

naturality :  {x y} {xy : x  y} {f g : A  B}
             (fg :  x  f x  g x) 
             trans (cong f xy) (fg y)  trans (fg x) (cong g xy)
naturality {x = x} {xy = refl} fg =
  fg x                sym (trans-refl _) 
  trans (fg x) refl  
  where open -Reasoning

-- A lemma that is very similar to Corollary2.4.4 from the HoTT book.

cong-id :  {f : A  A} {x : A} (fid :  x  f x  x) 
           cong f (fid x)  fid (f x)
cong-id {f = f} {x} fid = begin
  cong f fxx                                     sym (trans-refl _) 
  trans (cong f fxx) refl                        cong (trans _) (sym (trans-sym fxx)) 
  trans (cong f fxx) (trans fxx (sym fxx))     sym (trans-assoc (cong f fxx)) 
  trans (trans (cong f fxx) fxx) (sym fxx)     cong ( p  trans p (sym _)) (naturality fid) 
  trans (trans fxx (cong id fxx)) (sym fxx)   cong ( p  trans (trans fxx p) (sym fxx)) (cong-id _) 
  trans (trans fxx fxx) (sym fxx)             trans-assoc fxx 
  trans fxx (trans fxx (sym fxx))             cong (trans _) (trans-sym fxx) 
  trans fxx refl                                trans-refl _ 
  fid (f x)                                     
  where open -Reasoning; fxx = fid x; fxx = fid (f x)

-- Inspect

-- Inspect can be used when you want to pattern match on the result r
-- of some expression e, and you also need to "remember" that r  e.

-- See README.Inspect for an explanation of how/why to use this.

-- Normally (but not always) the new `with ... in` syntax described at
-- https://agda.readthedocs.io/en/v2.6.4/language/with-abstraction.html#with-abstraction-equality
-- can be used instead."

record Reveal__is_ {A : Set a} {B : A  Set b}
                    (f : (x : A)  B x) (x : A) (y : B x) :
                    Set (a  b) where
  constructor [_]
  field eq : f x  y

inspect :  {A : Set a} {B : A  Set b}
          (f : (x : A)  B x) (x : A)  Reveal f  x is f x
inspect f x = [ refl ]
