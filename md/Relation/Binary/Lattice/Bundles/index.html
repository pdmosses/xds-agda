    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Bundles for order-theoretic lattices
------------------------------------------------------------------------

-- The contents of this module should be accessed via
-- `Relation.Binary.Lattice`.

{-# OPTIONS --cubical-compatible --safe #-}

module Relation.Binary.Lattice.Bundles where

open import Algebra.Core
open import Level using (suc; __)
open import Relation.Binary.Core using (Rel)
open import Relation.Binary.Bundles using (Poset; Setoid)
open import Relation.Binary.Lattice.Structures

------------------------------------------------------------------------
-- Join semilattices

record JoinSemilattice c   : Set (suc (c    )) where
  infix  4 __ __
  infixr 6 __
  field
    Carrier           : Set c
    __               : Rel Carrier   -- The underlying equality.
    __               : Rel Carrier   -- The partial order.
    __               : Op Carrier     -- The join operation.
    isJoinSemilattice : IsJoinSemilattice __ __ __

  open IsJoinSemilattice isJoinSemilattice public

  poset : Poset c  
  poset = record { isPartialOrder = isPartialOrder }

  open Poset poset public using (preorder)


record BoundedJoinSemilattice c   : Set (suc (c    )) where
  infix  4 __ __
  infixr 6 __
  field
    Carrier                  : Set c
    __                      : Rel Carrier   -- The underlying equality.
    __                      : Rel Carrier   -- The partial order.
    __                      : Op Carrier     -- The join operation.
                            : Carrier         -- The minimum.
    isBoundedJoinSemilattice : IsBoundedJoinSemilattice __ __ __ 

  open IsBoundedJoinSemilattice isBoundedJoinSemilattice public

  joinSemilattice : JoinSemilattice c  
  joinSemilattice = record { isJoinSemilattice = isJoinSemilattice }

  open JoinSemilattice joinSemilattice public using (preorder; poset)

-- Lattices

record Lattice c   : Set (suc (c    )) where
  infix  4 __ __
  infixr 6 __
  infixr 7 __
  field
    Carrier   : Set c
    __       : Rel Carrier   -- The underlying equality.
    __       : Rel Carrier   -- The partial order.
    __       : Op Carrier     -- The join operation.
    __       : Op Carrier     -- The meet operation.
    isLattice : IsLattice __ __ __ __

  open IsLattice isLattice public

  setoid : Setoid c 
  setoid = record { isEquivalence = isEquivalence }

  joinSemilattice : JoinSemilattice c  
  joinSemilattice = record { isJoinSemilattice = isJoinSemilattice }

  meetSemilattice : MeetSemilattice c  
  meetSemilattice = record { isMeetSemilattice = isMeetSemilattice }

  open JoinSemilattice joinSemilattice public using (poset; preorder)

record DistributiveLattice c   : Set (suc (c    )) where
  infix  4 __ __
  infixr 6 __
  infixr 7 __
  field
    Carrier : Set c
    __     : Rel Carrier   -- The underlying equality.
    __     : Rel Carrier   -- The partial order.
    __     : Op Carrier     -- The join operation.
    __     : Op Carrier     -- The meet operation.
    isDistributiveLattice : IsDistributiveLattice __ __ __ __

  open IsDistributiveLattice isDistributiveLattice using (-distrib-) public
  open IsDistributiveLattice isDistributiveLattice using (isLattice)

  lattice : Lattice c  
  lattice = record { isLattice = isLattice }

  open Lattice lattice hiding (Carrier; __; __; __; __) public

record BoundedLattice c   : Set (suc (c    )) where
  infix  4 __ __
  infixr 6 __
  infixr 7 __
  field
    Carrier          : Set c
    __              : Rel Carrier   -- The underlying equality.
    __              : Rel Carrier   -- The partial order.
    __              : Op Carrier     -- The join operation.
    __              : Op Carrier     -- The meet operation.
                    : Carrier         -- The maximum.
                    : Carrier         -- The minimum.
    isBoundedLattice : IsBoundedLattice __ __ __ __  

  open IsBoundedLattice isBoundedLattice public

  boundedJoinSemilattice : BoundedJoinSemilattice c  
  boundedJoinSemilattice = record
    { isBoundedJoinSemilattice = isBoundedJoinSemilattice }

  boundedMeetSemilattice : BoundedMeetSemilattice c  
  boundedMeetSemilattice = record
    { isBoundedMeetSemilattice = isBoundedMeetSemilattice }

  lattice : Lattice c  
  lattice = record { isLattice = isLattice }

  open Lattice lattice public
    using (joinSemilattice; meetSemilattice; poset; preorder; setoid)

-- Heyting algebras (a bounded lattice with exponential operator)

record HeytingAlgebra c   : Set (suc (c    )) where
  infix  4 __ __
  infixr 5 __
  infixr 6 __
  infixr 7 __
  field
    Carrier          : Set c
    __              : Rel Carrier   -- The underlying equality.
    __              : Rel Carrier   -- The partial order.
    __              : Op Carrier     -- The join operation.
    __              : Op Carrier     -- The meet operation.
    __              : Op Carrier     -- The exponential operation.
                    : Carrier         -- The maximum.
                    : Carrier         -- The minimum.
    isHeytingAlgebra : IsHeytingAlgebra __ __ __ __ __  

  boundedLattice : BoundedLattice c  
  boundedLattice = record
    { isBoundedLattice = IsHeytingAlgebra.isBoundedLattice isHeytingAlgebra }

  open IsHeytingAlgebra isHeytingAlgebra
    using (exponential; transpose-; transpose-) public
  open BoundedLattice boundedLattice
    hiding (Carrier; __; __; __; __; ; ) public

------------------------------------------------------------------------
-- Boolean algebras (a specialized Heyting algebra)

record BooleanAlgebra c   : Set (suc (c    )) where
  infix  4 __ __
  infixr 6 __
  infixr 7 __
  infix 8 _
  field
    Carrier          : Set c
    __              : Rel Carrier   -- The underlying equality.
    __              : Rel Carrier   -- The partial order.
    __              : Op Carrier     -- The join operation.
    __              : Op Carrier     -- The meet operation.
    _               : Op Carrier     -- The negation operation.
                    : Carrier         -- The maximum.
                    : Carrier         -- The minimum.
    isBooleanAlgebra : IsBooleanAlgebra __ __ __ __ _  

  open IsBooleanAlgebra isBooleanAlgebra using (isHeytingAlgebra)

  heytingAlgebra : HeytingAlgebra c  
  heytingAlgebra = record { isHeytingAlgebra = isHeytingAlgebra }

  open HeytingAlgebra heytingAlgebra public
    hiding (Carrier; __; __; __; __; ; )
