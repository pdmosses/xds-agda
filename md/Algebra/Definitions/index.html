    Algebra.Definitions

    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Properties of functions, such as associativity and commutativity
------------------------------------------------------------------------

-- The contents of this module should be accessed via `Algebra`, unless
-- you want to parameterise it via the equality relation.

-- Note that very few of the element arguments are made implicit here,
-- as we do not assume that the Agda can infer either the right or left
-- argument of the binary operators. This is despite the fact that the
-- library defines most of its concrete operators (e.g. in
-- `Data.Nat.Base`) as being left-biased.

{-# OPTIONS --cubical-compatible --safe #-}

open import Relation.Binary.Core using (Rel; _Preserves__; _Preserves___)
open import Relation.Nullary.Negation.Core using (_)

module Algebra.Definitions
  {a } {A : Set a}   -- The underlying set
  (__ : Rel A )     -- The underlying equality
  where

open import Algebra.Core using (Op; Op)
open import Data.Product.Base using (__; -syntax)
open import Data.Sum.Base using (__)

------------------------------------------------------------------------
-- Properties of operations

Congruent : Op A  Set _
Congruent f = f Preserves __  __

Congruent : Op A  Set _
Congruent  =  Preserves __  __  __

LeftCongruent : Op A  Set _
LeftCongruent __ =  {x}  (x _) Preserves __  __

RightCongruent : Op A  Set _
RightCongruent __ =  {x}  (_ x) Preserves __  __

Associative : Op A  Set _
Associative __ =  x y z  ((x  y)  z)  (x  (y  z))

Commutative : Op A  Set _
Commutative __ =  x y  (x  y)  (y  x)

LeftIdentity : A  Op A  Set _
LeftIdentity e __ =  x  (e  x)  x

RightIdentity : A  Op A  Set _
RightIdentity e __ =  x  (x  e)  x

Identity : A  Op A  Set _
Identity e  = (LeftIdentity e )  (RightIdentity e )

LeftZero : A  Op A  Set _
LeftZero z __ =  x  (z  x)  z

RightZero : A  Op A  Set _
RightZero z __ =  x  (x  z)  z

Zero : A  Op A  Set _
Zero z  = (LeftZero z )  (RightZero z )

LeftInverse : A  Op A  Op A  Set _
LeftInverse e _ __ =  x  ((x )  x)  e

RightInverse : A  Op A  Op A  Set _
RightInverse e _ __ =  x  (x  (x ))  e

Inverse : A  Op A  Op A  Set _
Inverse e   = (LeftInverse e )   (RightInverse e  )

-- For structures in which not every element has an inverse (e.g. Fields)
LeftInvertible : A  Op A  A  Set _
LeftInvertible e __ x = [ x ] (x  x)  e

RightInvertible : A  Op A  A  Set _
RightInvertible e __ x = [ x ] (x  x)  e

-- NB: this is not quite the same as
-- LeftInvertible e  x  RightInvertible e  x
-- since the left and right inverses have to coincide.
Invertible : A  Op A  A  Set _
Invertible e __ x = [ x ] (x  x)  e  (x  x)  e

LeftConical : A  Op A  Set _
LeftConical e __ =  x y  (x  y)  e  x  e

RightConical : A  Op A  Set _
RightConical e __ =  x y  (x  y)  e  y  e

Conical : A  Op A  Set _
Conical e  = (LeftConical e )  (RightConical e )

infix 4 _DistributesOver_ _DistributesOver_ _DistributesOver_

_DistributesOver_ : Op A  Op A  Set _
_*_ DistributesOver _+_ =
   x y z  (x * (y + z))  ((x * y) + (x * z))

_DistributesOver_ : Op A  Op A  Set _
_*_ DistributesOver _+_ =
   x y z  ((y + z) * x)  ((y * x) + (z * x))

_DistributesOver_ : Op A  Op A  Set _
* DistributesOver + = (* DistributesOver +)  (* DistributesOver +)

infix 4 _MiddleFourExchange_ _IdempotentOn_ _Absorbs_

_MiddleFourExchange_ : Op A  Op A  Set _
_*_ MiddleFourExchange _+_ =
   w x y z  ((w + x) * (y + z))  ((w + y) * (x + z))

_IdempotentOn_ : Op A  A  Set _
__ IdempotentOn x = (x  x)  x

Idempotent : Op A  Set _
Idempotent  =  x   IdempotentOn x

IdempotentFun : Op A  Set _
IdempotentFun f =  x  f (f x)  f x

Selective : Op A  Set _
Selective __ =  x y  (x  y)  x  (x  y)  y

_Absorbs_ : Op A  Op A  Set _
__ Absorbs __ =  x y  (x  (x  y))  x

Absorptive : Op A  Op A  Set _
Absorptive   = ( Absorbs )  ( Absorbs )

SelfInverse : Op A  Set _
SelfInverse f =  {x y}  f x  y  f y  x

Involutive : Op A  Set _
Involutive f =  x  f (f x)  x

LeftCancellative : Op A  Set _
LeftCancellative __ =  x y z  (x  y)  (x  z)  y  z

RightCancellative : Op A  Set _
RightCancellative __ =  x y z  (y  x)  (z  x)  y  z

Cancellative : Op A  Set _
Cancellative __ = (LeftCancellative __)  (RightCancellative __)

AlmostLeftCancellative : A  Op A  Set _
AlmostLeftCancellative e __ =  x y z   x  e  (x  y)  (x  z)  y  z

AlmostRightCancellative : A  Op A  Set _
AlmostRightCancellative e __ =  x y z   x  e  (y  x)  (z  x)  y  z

AlmostCancellative : A  Op A  Set _
AlmostCancellative e __ = AlmostLeftCancellative e __  AlmostRightCancellative e __

Interchangable : Op A  Op A  Set _
Interchangable __ __ =  w x y z  ((w  x)  (y  z))  ((w  y)  (x  z))

LeftDivides : Op A  Op A  Set _
LeftDivides __  _\\_ =  x y  (x  (x \\ y))  y

LeftDivides : Op A  Op A  Set _
LeftDivides __ _\\_ =  x y  (x \\ (x  y))  y

RightDivides : Op A  Op A  Set _
RightDivides __ _//_ =  x y  ((y // x)  x)  y

RightDivides : Op A  Op A  Set _
RightDivides __ _//_ =  x y  ((y  x) // x)  y

LeftDivides : Op A  Op A  Set _
LeftDivides  \\ = (LeftDivides  \\)  (LeftDivides  \\)

RightDivides : Op A  Op A  Set _
RightDivides  // = (RightDivides  //)  (RightDivides  //)

StarRightExpansive : A  Op A  Op A  Op A  Set _
StarRightExpansive e _+_ __ _* =  x  (e + (x  (x *)))  (x *)

StarLeftExpansive : A  Op A  Op A  Op A  Set _
StarLeftExpansive e _+_ __ _* =  x   (e + ((x *)  x))  (x *)

StarExpansive : A  Op A  Op A  Op A  Set _
StarExpansive e _+_ __ _* = (StarLeftExpansive e _+_ __ _*)  (StarRightExpansive e _+_ __ _*)

StarLeftDestructive : Op A  Op A  Op A  Set _
StarLeftDestructive _+_ __ _* =  a b x  (b + (a  x))  x  ((a *)  b)  x

StarRightDestructive : Op A  Op A  Op A  Set _
StarRightDestructive _+_ __ _* =  a b x  (b + (x  a))  x  (b  (a *))  x

StarDestructive : Op A  Op A  Op A  Set _
StarDestructive _+_ __ _* = (StarLeftDestructive _+_ __ _*)  (StarRightDestructive _+_ __ _*)

LeftAlternative : Op A  Set _
LeftAlternative __ =  x y    ((x  x)  y)  (x  (x  y))

RightAlternative : Op A  Set _
RightAlternative __ =  x y  (x  (y  y))  ((x  y)  y)

Alternative : Op A  Set _
Alternative __ = (LeftAlternative __ )  (RightAlternative __)

Flexible : Op A  Set _
Flexible __ =  x y  ((x  y)  x)  (x  (y  x))

Medial : Op A  Set _
Medial __ =  x y u z  ((x  y)  (u  z))  ((x  u)  (y  z))

LeftSemimedial : Op A  Set _
LeftSemimedial __ =  x y z  ((x  x)  (y  z))  ((x  y)  (x  z))

RightSemimedial : Op A  Set _
RightSemimedial __ =  x y z  ((y  z)  (x  x))  ((y  x)  (z  x))

Semimedial : Op A  Set _
Semimedial __ = (LeftSemimedial __)  (RightSemimedial __)

LeftBol : Op A  Set _
LeftBol __ =  x y z  (x  (y  (x  z)))  ((x  (y  x))  z )

RightBol : Op A  Set _
RightBol __ =  x y z  (((z  x)  y)  x)  (z  ((x  y)  x))

MiddleBol : Op A  Op A   Op A   Set _
MiddleBol __ _\\_ _//_ =  x y z  (x  ((y  z) \\ x))  ((x // z)  (y \\ x))

Identical : Op A  Set _
Identical __ =  x y z  ((z  x)  (y  z))  (z  ((x  y)  z))
