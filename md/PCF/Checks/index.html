    {-# OPTIONS --rewriting --confluence-check #-}
open import Agda.Builtin.Equality
open import Agda.Builtin.Equality.Rewrite

module PCF.Checks where

open import Data.Bool.Base 
open import Agda.Builtin.Nat
open import Relation.Binary.PropositionalEquality.Core
  using (__; refl; cong-app)

open import PCF.Domain-Notation
open import PCF.Types
open import PCF.Constants
open import PCF.Variables
open import PCF.Environments
open import PCF.Terms

fix-app  :  {P D} (f : (P  D)  (P  D)) (p : P) 
              fix f p  f (fix f) p
fix-app  =  f  cong-app (fix-fix f) 

{-# REWRITE fix-app elim-- elim-- true-cond false-cond #-} 

-- Constants
pattern  n    =  (k n)
pattern succ   =  +1
pattern pred  =  -1
pattern if     =  
pattern       =  Y
pattern      =  Z

-- Variables
f  = var 0 
g  = var 1 (  )
h  = var 2 (    )
a  = var 3 
b  = var 4 

-- Arithmetic
check-41+1 :  succ   41     42
check-41+1 = refl

check-43-1 :  pred   43     42
check-43-1 = refl

-- Binding
check-id :  ( a   a)   42     42
check-id = refl

check-k :  ( a   b   a)   42   41     42
check-k = refl

check-ki :  ( a   b   b)   41   42     42
check-ki = refl

    \clearpage

    check-suc-41 :  ( a  (succ   a ))   41     42
check-suc-41 = refl

check-pred-42 :  ( a  (pred   a))   43     42
check-pred-42 = refl

check-if-zero :  if  (   0)   42   0     42
check-if-zero = refl

check-if-nonzero :  if  (   42)   0   42     42
check-if-nonzero = refl

-- fix (f. 42)  42
check-fix-const :
     ( f   42)   
    42
check-fix-const = fix-fix ( x   42)

-- fix (g. a. 42) 2  42
check-fix-lambda :
     ( g   a   42)   2   
    42
check-fix-lambda = refl

-- fix (g. a. ifz a then 42 else g (pred a)) 101  42
check-countdown :
     ( g   a 
              (if  (   a)   42  ( g  (pred   a))))
        101
      
    42
check-countdown = refl

-- fix (h. a. b. ifz a then b else h (pred a) (succ b)) 4 38  42
check-sum-42 :
   (  ( h   a   b 
              (if  (   a)   b  ( h  (pred   a)  (succ   b)))))
        4   38
      
    42
check-sum-42 = refl
