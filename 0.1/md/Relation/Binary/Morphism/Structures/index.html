-- The Agda standard library
--
-- Order morphisms
------------------------------------------------------------------------

{-# OPTIONS --cubical-compatible --safe #-}

open import Relation.Binary.Core using (Rel)

module Relation.Binary.Morphism.Structures
  {a b} {A : Set a} {B : Set b}
  where

open import Data.Product.Base using (_,_)
open import Function.Definitions using (Injective; Surjective; Bijective)
open import Level using (Level; __)

open import Relation.Binary.Morphism.Definitions A B

private
  variable
        : Level

------------------------------------------------------------------------
-- Relations
------------------------------------------------------------------------

record IsRelHomomorphism (__ : Rel A ) (__ : Rel B )
                         (_ : A  B) : Set (a    ) where
  field
    cong : Homomorphic __ __ _


record IsRelMonomorphism (__ : Rel A ) (__ : Rel B )
                         (_ : A  B) : Set (a    ) where
  field
    isHomomorphism : IsRelHomomorphism __ __ _
    injective      : Injective __ __ _

  open IsRelHomomorphism isHomomorphism public


record IsRelIsomorphism (__ : Rel A ) (__ : Rel B )
                        (_ : A  B) : Set (a  b    ) where
  field
    isMonomorphism : IsRelMonomorphism __ __ _
    surjective     : Surjective __ __ _

  open IsRelMonomorphism isMonomorphism public

  bijective : Bijective __ __ _
  bijective = injective , surjective


------------------------------------------------------------------------
-- Orders
------------------------------------------------------------------------

record IsOrderHomomorphism (__ : Rel A ) (__ : Rel B )
                           (__ : Rel A ) (__ : Rel B )
                           (_ : A  B) : Set (a        )
                           where
  field
    cong  : Homomorphic __ __ _
    mono  : Homomorphic __ __ _

  module Eq where
    isRelHomomorphism : IsRelHomomorphism __ __ _
    isRelHomomorphism = record { cong = cong }

  isRelHomomorphism : IsRelHomomorphism __ __ _
  isRelHomomorphism = record { cong = mono }


record IsOrderMonomorphism (__ : Rel A ) (__ : Rel B )
                           (__ : Rel A ) (__ : Rel B )
                           (_ : A  B) : Set (a        )
                           where
  field
    isOrderHomomorphism : IsOrderHomomorphism __ __ __ __ _
    injective           : Injective __ __ _
    cancel              : Injective __ __ _

  open IsOrderHomomorphism isOrderHomomorphism public
    hiding (module Eq)

  module Eq where
    isRelMonomorphism : IsRelMonomorphism __ __ _
    isRelMonomorphism = record
      { isHomomorphism = IsOrderHomomorphism.Eq.isRelHomomorphism isOrderHomomorphism
      ; injective      = injective
      }

  isRelMonomorphism : IsRelMonomorphism __ __ _
  isRelMonomorphism = record
    { isHomomorphism = isRelHomomorphism
    ; injective      = cancel
    }


record IsOrderIsomorphism (__ : Rel A ) (__ : Rel B )
                          (__ : Rel A ) (__ : Rel B )
                          (_ : A  B) : Set (a  b        )
                          where
  field
    isOrderMonomorphism : IsOrderMonomorphism __ __ __ __ _
    surjective          : Surjective __ __ _

  open IsOrderMonomorphism isOrderMonomorphism public
    hiding (module Eq)

  module Eq where
    isRelIsomorphism : IsRelIsomorphism __ __ _
    isRelIsomorphism = record
      { isMonomorphism = IsOrderMonomorphism.Eq.isRelMonomorphism isOrderMonomorphism
      ; surjective     = surjective
      }
