-- Some other properties

-- I took the statement of this lemma (called Uniqueness of
-- Complements) from some course notes, "Boolean Algebra", written
-- by Gert Smolka.

private
  lemma :  x y  x  y    x  y     x  y
  lemma x y xy= xy= = begin
     x                 -identity _ 
     x               -cong xy= 
     x  (x  y)        -distrib- _ _ _ 
     x  x   x  y    -cong $ -complement _ 
       x  y         -cong xy= 
    x  y   x  y     -distrib- _ _ _ 
    (x   x)  y        -cong $ -complement _ 
      y                -identity _ 
    y                  

 :    
 = lemma   (-identity _) (-zero _)

 :    
 = lemma   (-zero _) (-identity _)

-involutive : Involutive _
-involutive x = lemma ( x) x (-complement _) (-complement _)

deMorgan :  x y   (x  y)   x   y
deMorgan x y = lemma (x  y) ( x   y) lem lem
  where
  lem = begin
    (x  y)  ( x   y)           -distrib- _ _ _ 
    (x  y)   x  (x  y)   y   -cong $ -cong $ -comm _ _ 
    (y  x)   x  (x  y)   y   -assoc _ _ _  -cong  -assoc _ _ _ 
    y  (x   x)  x  (y   y)   (-cong $ -complement _)  -cong 
                                      (-cong $ -complement _) 
    (y  )  (x  )               -zero _  -cong  -zero _ 
                                 -identity _ 
                                  

  lem = begin
    (x  y)   x           -distrib- _ _ _ 
    (x   x)  (y   x)   -cong $ -complement _ 
      (y   x)           -identity _ 
    y   x                 -comm _ _ 
     x  y                

  lem = begin
    (x  y)  ( x   y)   -assoc _ _ _ 
    ((x  y)   x)   y   -cong lem 
    ( x  y)   y         -assoc _ _ _ 
     x  (y   y)         -cong $ -complement _ 
     x                   -zero _ 
                          

deMorgan :  x y   (x  y)   x   y
deMorgan x y = begin
   (x  y)           -cong $ ((-involutive _)  -cong  (-involutive _)) 
   (  x    y)   -cong $ deMorgan _ _ 
    ( x   y)     -involutive _ 
   x   y

-- (, , id, , ) is a commutative ring

-- This construction is parameterised over the definition of xor.

module XorRing
  (xor : Op Carrier)
  (-def :  x y  xor x y  (x  y)   (x  y))
  where

  private
    infixl 6 __

    __ : Op Carrier
    __ = xor

    helper :  {x y u v}  x  y  u  v  x   u  y   v
    helper xy uv = xy  -cong  -cong uv

  -cong : Congruent __
  -cong {x} {y} {u} {v} xy uv = begin
    x  u                  -def _ _ 
    (x  u)   (x  u)    helper (xy  -cong  uv)
                                    (xy  -cong  uv) 
    (y  v)   (y  v)   -def _ _ 
    y  v                

  -comm : Commutative __
  -comm x y = begin
    x  y                  -def _ _ 
    (x  y)   (x  y)    helper (-comm _ _) (-comm _ _) 
    (y  x)   (y  x)   -def _ _ 
    y  x                

  -distrib- :  x y   (x  y)   x  y
  -distrib- x y = begin
     (x  y)                               -cong $ -def _ _ 
     ((x  y)  ( (x  y)))               -cong (-distrib- _ _ _) 
     ((x   (x  y))  (y   (x  y)))   -cong $ -cong $ -cong $ -cong (-comm _ _) 
     ((x   (x  y))  (y   (y  x)))   -cong $ lem _ _  -cong  lem _ _ 
     ((x   y)  (y   x))               deMorgan _ _ 
     (x   y)   (y   x)               -cong $ deMorgan _ _ 
    ( x  (  y))   (y   x)           helper (-cong $ -involutive _) (-comm _ _) 
    ( x  y)   ( x  y)                 -def _ _ 
     x  y                                
    where
    lem :  x y  x   (x  y)  x   y
    lem x y = begin
      x   (x  y)           -cong $ deMorgan _ _ 
      x  ( x   y)         -distrib- _ _ _ 
      (x   x)  (x   y)   -cong $ -complement _ 
        (x   y)           -identity _ 
      x   y                

  -distrib- :  x y   (x  y)  x   y
  -distrib- x y = begin
     (x  y)   -cong $ -comm _ _ 
     (y  x)   -distrib- _ _ 
     y  x     -comm _ _ 
    x   y    

  -annihilates- :  x y  x  y   x   y
  -annihilates- x y = begin
    x  y         -involutive _ 
      (x  y)    -cong $ -distrib- _ _ 
     ( x  y)    -distrib- _ _ 
     x   y    

  -identity : LeftIdentity  __
  -identity x = begin
      x                 -def _ _ 
    (  x)   (  x)   helper (-identity _) (-zero _) 
    x                  -cong  
    x                   -identity _ 
    x                    

  -identity : RightIdentity  __
  -identity _ = -comm _ _  trans  -identity _

  -identity : Identity  __
  -identity = -identity , -identity

  -inverse : LeftInverse  id __
  -inverse x = begin
    x  x                -def _ _ 
    (x  x)   (x  x)  helper (-idem _) (-idem _) 
    x   x              -complement _ 
                       

  -inverse : RightInverse  id __
  -inverse _ = -comm _ _  trans  -inverse _

  -inverse : Inverse  id __
  -inverse = -inverse , -inverse

  -distrib- : __ DistributesOver __
  -distrib- x y z = begin
    x  (y  z)                 -cong $ -def _ _ 
    x  ((y  z)   (y  z))   -assoc _ _ _ 
    (x  (y  z))   (y  z)   -cong $ deMorgan _ _ 
    (x  (y  z)) 
    ( y   z)                 -identity _ 
     
    ((x  (y  z)) 
    ( y   z))                -cong lem 
    ((x  (y  z))   x) 
    ((x  (y  z)) 
    ( y   z))                -distrib- _ _ _ 
    (x  (y  z)) 
    ( x  ( y   z))         -cong $ -cong (deMorgan _ _) 
    (x  (y  z)) 
    ( x   (y  z))           -cong (deMorgan _ _) 
    (x  (y  z)) 
     (x  (y  z))             helper refl lem 
    (x  (y  z)) 
     ((x  y)  (x  z))       -cong $ -distrib- _ _ _ 
    ((x  y)  (x  z)) 
     ((x  y)  (x  z))       -def _ _ 
    (x  y)  (x  z)          
      where
      lem = begin
        x  (y  z)   -assoc _ _ _ 
        (x  y)  z   -cong $ -comm _ _ 
        (y  x)  z   -assoc _ _ _ 
        y  (x  z)  

      lem = begin
        x  (y  z)         -cong (-idem _) 
        (x  x)  (y  z)   -assoc _ _ _ 
        x  (x  (y  z))   -cong lem 
        x  (y  (x  z))   -assoc _ _ _ 
        (x  y)  (x  z)  

      lem = begin
                               -zero _ 
        (y  z)               -cong (-complement _) 
        (y  z)  (x   x)     -assoc _ _ _ 
        ((y  z)  x)   x     -cong (-comm _ _) 
        (x  (y  z))   x    

  -distrib- : __ DistributesOver __
  -distrib- = commdistrdistr -cong -comm -distrib-

  -distrib- : __ DistributesOver __
  -distrib- = -distrib- , -distrib-

  private

    lemma :  x y u v 
             (x  y)  (u  v) 
             ((x  u)  (y  u)) 
             ((x  v)  (y  v))
    lemma x y u v = begin
        (x  y)  (u  v)               -distrib- _ _ _ 
        ((x  y)  u)  ((x  y)  v)   -distrib- _ _ _
                                             -cong 
                                          -distrib- _ _ _ 
        ((x  u)  (y  u)) 
        ((x  v)  (y  v))            

  -assoc : Associative __
  -assoc x y z = sym $ begin
    x  (y  z)                                 -cong refl (-def _ _) 
    x  ((y  z)   (y  z))                   -def _ _ 
      (x  ((y  z)   (y  z))) 
     (x  ((y  z)   (y  z)))               -cong lem lem 
    (((x  y)  z)  ((x   y)   z)) 
    ((( x   y)  z)  (( x  y)   z))     -assoc _ _ _ 
    ((x  y)  z) 
    (((x   y)   z) 
     ((( x   y)  z)  (( x  y)   z)))   -cong lem 
    ((x  y)  z) 
    ((( x   y)  z) 
     (((x   y)   z)  (( x  y)   z)))   -assoc _ _ _ 
    (((x  y)  z)  (( x   y)  z)) 
    (((x   y)   z)  (( x  y)   z))     -cong lem lem 
      (((x  y)   (x  y))  z) 
     (((x  y)   (x  y))  z)               -def _ _ 
    ((x  y)   (x  y))  z                   -cong (-def _ _) refl 
    (x  y)  z                                
    where
    lem = begin
      ((x  y)  z)  (( x   y)  z)   -distrib- _ _ _ 
      ((x  y)  ( x   y))  z         -cong $ -cong (deMorgan _ _) 
      ((x  y)   (x  y))  z          

    lem = begin
      (x   y)  ( x  y)               -cong (-identity _) (-identity _) 
      (  (x   y))  (( x  y)  )   -cong
                                              (-cong (-complement _) (-comm _ _))
                                              (-cong $ -complement _) 
      (( x  x)  ( y  x)) 
      (( x  y)  ( y  y))             lemma _ _ _ _ 
      ( x   y)  (x  y)               -cong (deMorgan _ _) (-involutive _) 
       (x  y)    (x  y)             deMorgan _ _ 
       ((x  y)   (x  y))            

    lem = begin
      ((x   y)   z)  (( x  y)   z)   -distrib- _ _ _ 
      ((x   y)  ( x  y))   z           -cong lem 
       ((x  y)   (x  y))   z           deMorgan _ _ 
       (((x  y)   (x  y))  z)          

    lem = begin
      x  ((y  z)   (y  z))           -cong $ -cong $ deMorgan _ _ 
      x  ((y  z)  ( y   z))         -distrib- _ _ _ 
      (x  (y  z))  (x  ( y   z))   -assoc _ _ _  -cong  -assoc _ _ _ 
      ((x  y)  z)  ((x   y)   z)  

    lem = begin
       ((y  z)   (y  z))     deMorgan _ _ 
       (y  z)    (y  z)     deMorgan _ _  -cong  -involutive _ 
      ( y   z)  (y  z)       lemma _ _ _ _ 
      (( y  y)  ( z  y)) 
      (( y  z)  ( z  z))     (-complement _  -cong  -comm _ _)
                                       -cong 
                                   (-cong $ -complement _) 
      (  (y   z)) 
      (( y  z)  )             -identity _  -cong  -identity _ 
      (y   z)  ( y  z)      

    lem = begin
       (x  ((y  z)   (y  z)))   deMorgan _ _ 
       x   ((y  z)   (y  z))   -cong lem 
       x  ((y   z)  ( y  z))   -distrib- _ _ _ 
      ( x  (y       z)) 
      ( x  ( y  z))               -assoc _ _ _  -cong  -assoc _ _ _ 
      (( x  y)       z) 
      (( x   y)  z)               -comm _ _ 
      (( x   y)  z) 
      (( x  y)       z)          

    lem = begin
      ((x   y)   z) 
      ((( x   y)  z)  (( x  y)   z))     -assoc _ _ _ 
      (((x   y)   z)  (( x   y)  z)) 
      (( x  y)   z)                           -cong $ -comm _ _ 
      ((( x   y)  z)  ((x   y)   z)) 
      (( x  y)   z)                           -assoc _ _ _ 
      (( x   y)  z) 
      (((x   y)   z)  (( x  y)   z))
