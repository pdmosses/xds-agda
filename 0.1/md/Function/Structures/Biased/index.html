    Function.Structures.Biased

    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Ways to give instances of certain structures where some fields can
-- be given in terms of others.
-- The contents of this file should usually be accessed from `Function`.
------------------------------------------------------------------------


{-# OPTIONS --cubical-compatible --safe #-}

open import Relation.Binary.Core using (Rel)


module Function.Structures.Biased {a b  }
  {A : Set a} (__ : Rel A ) -- Equality over the domain
  {B : Set b} (__ : Rel B ) -- Equality over the codomain
  where

open import Data.Product.Base as Product using (; __; _,_)
open import Function.Base using (__; id)
open import Function.Definitions using(StrictlySurjective; StrictlyInverse; StrictlyInverse; Congruent)
open import Function.Consequences.Setoid
  using (strictlySurjectivesurjective; strictlyInverseinverse
        ; strictlyInverseinverse)
open import Level using (__)
open import Relation.Binary.Bundles using (Setoid)
open import Relation.Binary.Structures using (IsEquivalence)

open import Function.Structures __ __

------------------------------------------------------------------------
-- Surjection
------------------------------------------------------------------------

record IsStrictSurjection (f : A  B) : Set (a  b    ) where
  field
    isCongruent : IsCongruent f
    strictlySurjective : StrictlySurjective __ f

  open IsCongruent isCongruent public

  isSurjection : IsSurjection f
  isSurjection = record
    { isCongruent = isCongruent
    ; surjective = strictlySurjectivesurjective
        Eq.setoid Eq.setoid cong strictlySurjective
    }

open IsStrictSurjection public
  using () renaming (isSurjection to isStrictSurjection)

------------------------------------------------------------------------
-- Bijection
------------------------------------------------------------------------

record IsStrictBijection (f : A  B) : Set (a  b    ) where
  field
    isInjection : IsInjection f
    strictlySurjective  : StrictlySurjective __ f

  isBijection : IsBijection f
  isBijection = record
    { isInjection = isInjection
    ; surjective = strictlySurjectivesurjective
        Eq.setoid Eq.setoid cong strictlySurjective
    } where open IsInjection isInjection

open IsStrictBijection public
  using () renaming (isBijection to isStrictBijection)

------------------------------------------------------------------------
-- Left inverse
------------------------------------------------------------------------

record IsStrictLeftInverse (to : A  B) (from : B  A) : Set (a  b    ) where
  field
    isCongruent  : IsCongruent to
    from-cong    : Congruent __ __ from
    strictlyInverse : StrictlyInverse __ to from

  isLeftInverse : IsLeftInverse to from
  isLeftInverse = record
    { isCongruent = isCongruent
    ; from-cong = from-cong
    ; inverse = strictlyInverseinverse
        Eq.setoid Eq.setoid cong strictlyInverse
    } where open IsCongruent isCongruent

open IsStrictLeftInverse public
  using () renaming (isLeftInverse to isStrictLeftInverse)

------------------------------------------------------------------------

    strictlyInverse : StrictlyInverse __ to from

  isRightInverse : IsRightInverse to from
  isRightInverse = record
    { isCongruent = isCongruent
    ; from-cong = from-cong

open IsStrictRightInverse public

------------------------------------------------------------------------
-- Inverse
------------------------------------------------------------------------

record IsStrictInverse (to : A  B) (from : B  A) : Set (a  b    ) where
  field
    isLeftInverse : IsLeftInverse to from
    strictlyInverse : StrictlyInverse __ to from

  isInverse : IsInverse to from
  isInverse = record
    { isLeftInverse = isLeftInverse
    ; inverse      = strictlyInverseinverse
        Eq.setoid Eq.setoid from-cong strictlyInverse
    } where open IsLeftInverse isLeftInverse

open IsStrictInverse public
  using () renaming (isInverse to isStrictInverse)
