    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Some theory for Semigroup
------------------------------------------------------------------------

{-# OPTIONS --cubical-compatible --safe #-}

open import Algebra using (Semigroup)

module Algebra.Properties.Semigroup {a } (S : Semigroup a ) where

open import Data.Product.Base using (_,_)

open Semigroup S
  using (Carrier; __; __; setoid; trans ; refl; sym; assoc; -cong; -cong; -cong)
open import Algebra.Definitions __
  using (Alternative; LeftAlternative; RightAlternative; Flexible)

private
  variable
    u v w x y z : Carrier

xyzxyz :  x y z  x  (y  z)  (x  y)  z
xyzxyz x y z = sym (assoc x y z)

alternative : LeftAlternative __
alternative x y = assoc x x y

alternative : RightAlternative __
alternative x y = sym (assoc x y y)

alternative : Alternative __
alternative = alternative , alternative

flexible : Flexible __
flexible x y = assoc x y x

module _ (uvw : u  v  w) where
  uvwxuvxw :  x  (x  u)  v  x  w
  uvwxuvxw x = trans (assoc x u v) (-cong uvw)

  uvwuvxwx :  x  u  (v  x)  w  x
  uvwuvxwx x = trans (sym (assoc u v x)) (-cong uvw)

  uvwu[vxy]wxy :  x y  u  ((v  x)  y)  w  (x  y)
  uvwu[vxy]wxy x y = trans (-cong (assoc v x y)) (uvwuvxwx (x  y))

  uvwx[uvy]xwy :  x y  x  (u  (v  y))  x  (w  y)
  uvwx[uvy]xwy x y = -cong (uvwuvxwx y)

  uvw[xyu]vxyw :  x y  (x  (y  u))  v  x  (y  w)
  uvw[xyu]vxyw x y = trans (assoc x (y  u) v) (-cong (uvwxuvxw y))

  uvw[xuv]yxwy :  x y  ((x  u)  v)  y  x  (w  y)
  uvw[xuv]yxwy x y = trans (-cong (uvwxuvxw x)) (assoc x w y)

  uvw[xyu]vxyw :  x y  ((x  y)  u)  v  x  (y  w)
  uvw[xyu]vxyw x y = trans (-cong (assoc x y u)) (uvw[xyu]vxyw x y )

module _ (uvw : u  v  w) where

  uvwxuvyxwy :  x y  (x  u)  (v  y)  x  (w  y)
  uvwxuvyxwy x y = uvwxuvxw (uvwuvxwx uvw y) x

  uvwxyzu[vxy]wz :  x y  x  y  z  u  ((v  x)  y)  w  z
  uvwxyzu[vxy]wz x y xyz = trans (-cong (uvwxuvxw xyz v)) (uvwuvxwx uvw _)

  uvwxwyx[uvy] : x  (w  y)  x  (u  (v  y))
  uvwxwyx[uvy] = sym (uvwx[uvy]xwy uvw _ _)

module _ u v w x where
  [uvw]xu[vwx] : ((u  v)  w)  x  u  ((v  w)  x)
  [uvw]xu[vwx] = uvw[xuv]yxwy refl u x

  [uvw]xu[vwx] : ((u  v)  w)  x  u  (v  (w  x))
  [uvw]xu[vwx] = uvw[xyu]vxyw refl u v

  [uvw]xuvwx : (u  (v  w))  x  (u  v)  (w  x)
  [uvw]xuvwx = trans (sym (-cong (assoc u v w))) (assoc (u  v) w x)

  [uvw]xu[vwx] : (u  (v  w))  x  u  (v  (w  x))
  [uvw]xu[vwx] = uvw[xyu]vxyw refl u v

  uvwxu[vwx] : (u  v)  (w  x)  u  ((v  w)  x)
  uvwxu[vwx] = uvwxuvyxwy refl u x

module _ (uvwx : u  v  w  x) where
  uvwxyuvywx :  y  (y  u)  v  (y  w)  x
  uvwxyuvywx y = trans (uvwxuvxw uvwx y) (sym (assoc y w x))

  uvwxuvywxy :  y  u  (v  y)  w  (x  y)
  uvwxuvywxy y = trans (uvwuvxwx uvwx y) (assoc w x y)

  uvwxyuvzywxz :  y z  (y  u)  (v  z)  (y  w)  (x  z)
  uvwxyuvzywxz y z = trans (uvwxuvxw (uvwxuvywxy z) y) (sym (assoc y w (x  z)))
