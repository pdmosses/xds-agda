    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Properties of min and max operators specified over a total
-- preorder.
------------------------------------------------------------------------

{-# OPTIONS --cubical-compatible --safe #-}

open import Algebra.Construct.NaturalChoice.Base using (MinOperator; MaxOperator)
open import Relation.Binary.Bundles using (TotalPreorder)

module Algebra.Construct.NaturalChoice.MinMaxOp
  {a  } {O : TotalPreorder a  }
  (minOp : MinOperator O)
  (maxOp : MaxOperator O)
  where

open import Data.Sum.Base as Sum using (inj; inj; [_,_])
open import Data.Product.Base using (_,_)
open import Function.Base using (id; __; flip)
open import Relation.Binary.Core using (_Preserves__)

open TotalPreorder O renaming
  ( Carrier   to A
  ; __       to __
  ; -resp-  to -resp-
  ; -resp- to -resp-
  ; -resp- to -resp-
  )
open MinOperator minOp
open MaxOperator maxOp

open import Algebra.Definitions __
open import Algebra.Structures __
open import Algebra.Consequences.Setoid Eq.setoid
open import Relation.Binary.Reasoning.Preorder preorder

------------------------------------------------------------------------
-- Re-export properties of individual operators

open import Algebra.Construct.NaturalChoice.MinOp minOp public
open import Algebra.Construct.NaturalChoice.MaxOp maxOp public

------------------------------------------------------------------------
-- Joint algebraic structures

-distrib- : __ DistributesOver __
-distrib- x y z with total y z
... | inj yz = begin-equality
  x  (y  z)         -cong x (xyxyy yz) 
  x  z              xyxyy (-mono- x yz) 
  (x  y)  (x  z) 
... | inj yz = begin-equality
  x  (y  z)         -cong x (xyxyx yz) 
  x  y              xyxyx (-mono- x yz) 
  (x  y)  (x  z) 

-distrib- : __ DistributesOver __
-distrib- = commdistrdistr -cong -comm -distrib-

-distrib- : __ DistributesOver __
-distrib- = -distrib- , -distrib-

-absorbs- : __ Absorbs __
-absorbs- x y with total x y
... | inj xy = begin-equality
  x  (x  y)   -cong x (xyxyy xy) 
  x  y         xyxyx xy 
  x            
... | inj yx = begin-equality
  x  (x  y)   -cong x (xyxyx yx) 
  x  x         -idem x 
  x

antimono--distrib- :  {f}  f Preserves __  __  f Preserves __  __ 
                        x y  f (x  y)  f x  f y
antimono--distrib- {f} cong antimono x y with total x y
... | inj xy = begin-equality
  f (x  y)   cong (xyxyx xy) 
  f x         xyxyx (antimono xy) 
  f x  f y  
... | inj yx = begin-equality
  f (x  y)   cong (xyxyy yx) 
  f y         xyxyy (antimono yx) 
  f x  f y
