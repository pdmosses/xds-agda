    Notation

    {-# OPTIONS --rewriting --confluence-check --lossy-unification #-}
open import Agda.Builtin.Equality
open import Agda.Builtin.Equality.Rewrite

module Notation where

open import Data.Bool.Base  using (Bool; false; true) public
open import Data.Nat.Base   renaming ( to Nat) using (suc) public
open import Function        using (id; __) public

postulate
  Domain  : Set          -- type of all domains
  _     : Domain  Set  -- carrier of a domain

variable
  A B C   : Set
  D E F   : Domain
  n       : Nat

postulate
         :  D          -- bottom element
         : Domain        -- trivial domain

    Function domains

    module Functions where

  postulate
    __     : Domain  Domain  Domain -- assume continuous
    dom-cts  :  D  E   ( D    E )

  {-# REWRITE dom-cts #-}

  infixr 0 __

  postulate
    fix :  (D  D)  D  -- fixed points of endofunctions

  postulate
    __     : Set  Domain  Domain -- always continuous
    set-cts  :  A  E   (A   E )

  {-# REWRITE set-cts #-}

  infixr 0 __

open Functions public

    Lifted domains

    module Lifted where

  postulate
    _+  : Set  Domain               -- lifted set
        :  A  A +               -- inclusion
    _   :  (A  D)  A +  D   -- Kleisli extension

  infix 10 _+

    Booleans

      module Booleans where

    Bool = Bool +                      -- truth-value domain

    postulate
      __,_  :  Bool  D  D  D   -- McCarthy conditional

    infixr 20 __,_

    Naturals

      module Naturals where

    Nat = Nat + -- natural number domain

    open Booleans

    postulate
      _==_  :  Nat  Nat  Bool  -- strict numerical equality

    Strings

      module Strings where
    open import Data.String.Base using (String) public

    String = String + -- meta-string domain

    Sum domains

    module Sums where

  postulate
    _+_    : Domain  Domain  Domain                 -- coalesced sum
    inj   :  D  D + E                             -- injection
    inj   :  E  D + E                             -- injection
    [_,_]  :  (D  F)  (E  F)  (D + E  F)   -- case analysis

  infixr 1 _+_

    Product domains

    module Products where

  postulate
    __   : Domain  Domain  Domain   -- cartesian product
    _,_   :  D  E  D  E          -- pairing
    _1  :  D  E  D               -- 1st projection
    _2  :  D  E  E               -- 2nd projection
    _1  :  D  E  F  D           -- 1st projection
    _2  :  D  E  F  E           -- 2nd projection
    _3  :  D  E  F  F           -- 3rd projection

  infixr 2 __
  infixr 4 _,_

    Tuples

      module Tuples where

    _^_ : Domain  Nat  Domain
    D ^ 0            =  
    D ^ 1            = D
    D ^ suc (suc n)  = D  (D ^ suc n)

    infix 8  _^_

    Sequences

      module Sequences where

    open Lifted.Naturals
    open Tuples

    postulate
      _     : Domain  Domain         -- D           finite sequences 
           :  D                   --             empty sequence
      _    :  (D ^ suc n)  D    --  d , ...   non-empty sequence
      #      :  D   Nat          -- # d          sequence length
      __    :  D   D   D     -- d  d       concatenation
      __    :  D   Nat  D      -- d  n        nth component
      __    :  D   Nat  D     -- d  n        nth tail

  open Strings   public
