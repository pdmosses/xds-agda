    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Properties of binary relations
------------------------------------------------------------------------

-- The contents of this module should be accessed via `Relation.Binary`.

{-# OPTIONS --cubical-compatible --safe #-}

module Relation.Binary.Definitions where

open import Agda.Builtin.Equality using (__)
open import Data.Product.Base using (__; -syntax)
open import Data.Sum.Base using (__)
open import Function.Base using (_on_; flip)
open import Level using (Level; __; suc)
open import Relation.Binary.Core
open import Relation.Nullary as Nullary using (_; Dec)

private
  variable
    a b c     : Level
    A : Set a
    B : Set b
    C : Set c

------------------------------------------------------------------------
-- Definitions
------------------------------------------------------------------------

-- Reflexivity - defined without an underlying equality. It could
-- alternatively be defined as `__  __` for some equality `__`.

-- Confusingly the convention in the library is to use the name "refl"
-- for proofs of Reflexive and `reflexive` for proofs of type `__  __`,
-- e.g. in the definition of `IsEquivalence` later in this file. This
-- convention is a legacy from the early days of the library.

Reflexive : Rel A   Set _
Reflexive __ =  {x}  x  x

-- Generalised symmetry.

Sym : REL A B   REL B A   Set _
Sym P Q = P  flip Q

-- Symmetry.

Symmetric : Rel A   Set _
Symmetric __ = Sym __ __

-- Generalised transitivity.

Trans : REL A B   REL B C   REL A C   Set _
Trans P Q R =  {i j k}  P i j  Q j k  R i k

RightTrans : REL A B   REL B B   Set _
RightTrans R S = Trans R S R

LeftTrans : REL A A   REL A B   Set _
LeftTrans S R = Trans S R R

-- A flipped variant of generalised transitivity.

TransFlip : REL A B   REL B C   REL A C   Set _
TransFlip P Q R =  {i j k}  Q j k  P i j  R i k

-- Transitivity.

Transitive : Rel A   Set _
Transitive __ = Trans __ __ __

-- Generalised antisymmetry

Antisym : REL A B   REL B A   REL A B   Set _
Antisym R S E =  {i j}  R i j  S j i  E i j

-- Antisymmetry.

Antisymmetric : Rel A   Rel A   Set _
Antisymmetric __ __ = Antisym __ __ __

-- Irreflexivity - this is defined terms of the underlying equality.

Irreflexive : REL A B   REL A B   Set _
Irreflexive __ _<_ =  {x y}  x  y   (x < y)

-- Asymmetry.

Asymmetric : Rel A   Set _
Asymmetric _<_ =  {x y}  x < y   (y < x)

-- Density

Dense : Rel A   Set _
Dense _<_ =  {x y}  x < y  [ z ] x < z  z < y

-- Generalised connex - at least one of the two relations holds.

Connex : REL A B   REL B A   Set _
Connex P Q =  x y  P x y  Q y x

-- Totality.

Total : Rel A   Set _
Total __ = Connex __ __

-- Generalised trichotomy - exactly one of three types has a witness.

data Tri (A : Set a) (B : Set b) (C : Set c) : Set (a  b  c) where
  tri< : ( a :   A) (b :  B) (c :  C)  Tri A B C
  tri : (a :  A) ( b :   B) (c :  C)  Tri A B C
  tri> : (a :  A) (b :  B) ( c :   C)  Tri A B C

-- Trichotomy.

Trichotomous : Rel A   Rel A   Set _
Trichotomous __ _<_ =  x y  Tri (x < y) (x  y) (x > y)
  where _>_ = flip _<_

-- Generalised maximum element.

Max : REL A B   B  Set _
Max __ T =  x  x  T

-- Maximum element.

Maximum : Rel A   A  Set _
Maximum = Max

-- Generalised minimum element.

Min : REL A B   A  Set _
Min R = Max (flip R)

-- Minimum element.

Minimum : Rel A   A  Set _
Minimum = Min

-- Definitions for apartness relations

-- Note that Cotransitive's arguments are permuted with respect to Transitive's.
Cotransitive : Rel A   Set _
Cotransitive _#_ =  {x y}  x # y   z  (x # z)  (z # y)

Tight : Rel A   Rel A   Set _
Tight __ _#_ =  x y  ( x # y  x  y)  (x  y   x # y)

-- Properties of order morphisms, aka order-preserving maps

Monotonic : Rel A   Rel B   (A  B)  Set _
Monotonic __ __ f = f Preserves __  __

Antitonic : Rel A   Rel B   (A  B)  Set _
Antitonic __ = Monotonic (flip __)

LeftMonotonic : Rel B   Rel C   (A  B  C)  Set _
LeftMonotonic __ __ __ =  x  Monotonic __ __ (x _)

RightMonotonic : Rel A   Rel C   (A  B  C)  Set _
RightMonotonic __ __ __ =  y  Monotonic __ __ (_ y)

Monotonic : Rel A   Rel B   Rel C   (A  B  C)  Set _
Monotonic __ __ __  =  Preserves __  __  __

MonotonicAntitonic : Rel A   Rel B   Rel C   (A  B  C)  Set _
MonotonicAntitonic __ __ = Monotonic __ (flip __)

AntitonicMonotonic : Rel A   Rel B   Rel C   (A  B  C)  Set _
AntitonicMonotonic __ = Monotonic (flip __)

Antitonic : Rel A   Rel B   Rel C   (A  B  C)  Set _
Antitonic __ __ = Monotonic (flip __) (flip __)

Adjoint : Rel A   Rel B   (A  B)  (B  A)  Set _
Adjoint __ __ f g =  {x y}  (f x  y  x  g y)  (x  g y  f x  y)

-- Unary relations respecting a binary relation.

__Respects_ : (A  Set )  (B  Set )  REL A B   Set _
P  Q Respects __ =  {x y}  x  y  P x  Q y

-- Unary relation respects a binary relation.

_Respects_ : (A  Set )  Rel A   Set _
P Respects __ = P  P Respects __

-- Right respecting - relatedness is preserved on the right by equality.

_Respects_ : REL A B   Rel B   Set _
__ Respects __ =  {x}  (x _) Respects __

-- Left respecting - relatedness is preserved on the left by equality.

_Respects_ : REL A B   Rel A   Set _
P Respects __ =  {y}  (flip P y) Respects __

-- Respecting - relatedness is preserved on both sides by equality

_Respects_ : Rel A   Rel A   Set _
P Respects __ = (P Respects __)  (P Respects __)

-- Substitutivity - any two related elements satisfy exactly the same
-- set of unary relations. Note that only the various derivatives
-- of propositional equality can satisfy this property.

Substitutive : Rel A   ( : Level)  Set _
Substitutive {A = A} __ p = (P : A  Set p)  P Respects __

-- Irrelevancy - all proofs that a given pair of elements are related
-- are indistinguishable.

Irrelevant : REL A B   Set _
Irrelevant __ =  {x y}  Nullary.Irrelevant (x  y)

-- Recomputability - we can rebuild a relevant proof given an
-- irrelevant one.

Recomputable : REL A B   Set _
Recomputable __ =  {x y}  Nullary.Recomputable (x  y)

-- Stability

Stable : REL A B   Set _
Stable __ =  x y  Nullary.Stable (x  y)

-- Weak decidability - it is sometimes possible to determine if a given
-- pair of elements are related.

WeaklyDecidable : REL A B   Set _
WeaklyDecidable __ =  x y  Nullary.WeaklyDecidable (x  y)

-- Decidability - it is possible to determine whether a given pair of
-- elements are related.

Decidable : REL A B   Set _
Decidable __ =  x y  Dec (x  y)

-- Propositional equality is decidable for the type.

DecidableEquality : (A : Set a)  Set _
DecidableEquality A = Decidable {A = A} __

-- Universal - all pairs of elements are related

Universal : REL A B   Set _
Universal __ =  x y  x  y

-- Empty - no elements are related

Empty : REL A B   Set _
Empty __ =  {x y}   (x  y)

-- Non-emptiness - at least one pair of elements are related.

record NonEmpty {A : Set a} {B : Set b}
                (T : REL A B ) : Set (a  b  ) where
  constructor nonEmpty
  field
    {x}   : A
    {y}   : B
    proof : T x y
