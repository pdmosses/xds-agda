    ------------------------------------------------------------------------
-- The Agda standard library
--
-- Structures for order-theoretic lattices
------------------------------------------------------------------------

-- The contents of this module should be accessed via
-- `Relation.Binary.Lattice`.

{-# OPTIONS --cubical-compatible --safe #-}

open import Relation.Binary.Core using (Rel)

module Relation.Binary.Lattice.Structures
 {a  } {A : Set a}
 (__ : Rel A ) -- The underlying equality.
 (__ : Rel A ) -- The partial order.
 where

open import Algebra.Core using (Op; Op)
open import Algebra.Definitions using (_DistributesOver_)
open import Data.Product.Base using (__; _,_)
open import Level using (suc; __)
open import Relation.Binary.Definitions using (Minimum; Maximum)
open import Relation.Binary.Lattice.Definitions
  using (Supremum; Infimum; Exponential)
open import Relation.Binary.Structures using (IsPartialOrder)

------------------------------------------------------------------------
-- Join semilattices

record IsJoinSemilattice (__ : Op A)    -- The join operation.
                         : Set (a    ) where
  field
    isPartialOrder : IsPartialOrder __ __
    supremum       : Supremum __ __

  xxy :  x y  x  (x  y)
  xxy x y = let pf , _ , _ = supremum x y in pf

  yxy :  x y  y  (x  y)
  yxy x y = let _ , pf , _ = supremum x y in pf

  -least :  {x y z}  x  z  y  z  (x  y)  z
  -least {x} {y} {z} = let _ , _ , pf = supremum x y in pf z

  open IsPartialOrder isPartialOrder public

record IsBoundedJoinSemilattice (__ : Op A)    -- The join operation.
                                (   : A)        -- The minimum.
                                : Set (a    ) where
  field
    isJoinSemilattice : IsJoinSemilattice __
    minimum           : Minimum __ 

  open IsJoinSemilattice isJoinSemilattice public

------------------------------------------------------------------------
-- Meet semilattices

record IsMeetSemilattice (__ : Op A)    -- The meet operation.
                         : Set (a    ) where
  field
    isPartialOrder : IsPartialOrder __ __
    infimum        : Infimum __ __

  xyx :  x y  (x  y)  x
  xyx x y = let pf , _ , _ = infimum x y in pf

  xyy :  x y  (x  y)  y
  xyy x y = let _ , pf , _ = infimum x y in pf

  -greatest :  {x y z}  x  y  x  z  x  (y  z)
  -greatest {x} {y} {z} = let _ , _ , pf = infimum y z in pf x

  open IsPartialOrder isPartialOrder public

record IsBoundedMeetSemilattice (__ : Op A)    -- The join operation.
                                (   : A)        -- The maximum.
                                : Set (a    ) where
  field
    isMeetSemilattice : IsMeetSemilattice __
    maximum           : Maximum __ 

  open IsMeetSemilattice isMeetSemilattice public

------------------------------------------------------------------------
-- Lattices

record IsLattice (__ : Op A)    -- The join operation.
                 (__ : Op A)    -- The meet operation.
                 : Set (a    ) where
  field
    isPartialOrder : IsPartialOrder __ __
    supremum       : Supremum __ __
    infimum        : Infimum __ __

  isJoinSemilattice : IsJoinSemilattice  __
  isJoinSemilattice = record
    { isPartialOrder = isPartialOrder
    ; supremum       = supremum
    }

  isMeetSemilattice : IsMeetSemilattice  __
  isMeetSemilattice = record
    { isPartialOrder = isPartialOrder
    ; infimum        = infimum
    }

  open IsJoinSemilattice isJoinSemilattice public
    using (xxy; yxy; -least)
  open IsMeetSemilattice isMeetSemilattice public
    using (xyx; xyy; -greatest)
  open IsPartialOrder isPartialOrder public

record IsDistributiveLattice (__ : Op A)    -- The join operation.
                             (__ : Op A)    -- The meet operation.
                             : Set (a    ) where
  field
    isLattice    : IsLattice __ __
    -distrib- : _DistributesOver_ __ __ __

  open IsLattice isLattice public

record IsBoundedLattice (__ : Op A)    -- The join operation.
                        (__ : Op A)    -- The meet operation.
                        (   : A)        -- The maximum.
                        (   : A)        -- The minimum.
                        : Set (a    ) where
  field
    isLattice : IsLattice __ __
    maximum   : Maximum __ 
    minimum   : Minimum __ 

  open IsLattice isLattice public

  isBoundedJoinSemilattice : IsBoundedJoinSemilattice  __ 
  isBoundedJoinSemilattice = record
    { isJoinSemilattice = isJoinSemilattice
    ; minimum           = minimum
    }

  isBoundedMeetSemilattice : IsBoundedMeetSemilattice __ 
  isBoundedMeetSemilattice = record
    { isMeetSemilattice = isMeetSemilattice
    ; maximum           = maximum
    }

------------------------------------------------------------------------
-- Heyting algebras (a bounded lattice with exponential operator)

record IsHeytingAlgebra (__ : Op A)    -- The join operation.
                        (__ : Op A)    -- The meet operation.
                        (__ : Op A)    -- The exponential operation.
                        (   : A)        -- The maximum.
                        (   : A)        -- The minimum.
                        : Set (a    ) where
  field
    isBoundedLattice : IsBoundedLattice __ __  
    exponential      : Exponential __ __ __

  transpose- :  {w x y}  (w  x)  y  w  (x  y)
  transpose- {w} {x} {y} = let pf , _ = exponential w x y in pf

  transpose- :  {w x y}  w  (x  y)  (w  x)  y
  transpose- {w} {x} {y} = let _ , pf = exponential w x y in pf

  open IsBoundedLattice isBoundedLattice public

------------------------------------------------------------------------
-- Boolean algebras (a specialized Heyting algebra)

record IsBooleanAlgebra (__ : Op A)    -- The join operation.
                        (__ : Op A)    -- The meet operation.
                        (_ : Op A)     -- The negation operation.
                        (   : A)        -- The maximum.
                        (   : A)        -- The minimum.
                        : Set (a    ) where
  infixr 5 __
  __ : Op A
  x  y = ( x)  y

  field
    isHeytingAlgebra : IsHeytingAlgebra __ __ __  

  open IsHeytingAlgebra isHeytingAlgebra public
